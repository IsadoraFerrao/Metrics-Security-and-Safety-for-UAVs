58 static void md2_update_chksum(hash_state *md)
59 {
60  int j;
61  unsigned char L;
62  L = md->md2.chksum[15];
63  for (j = 0; j < 16; j++) {
64 
65 /* caution, the RFC says its "C[j] = S[M[i*16+j] xor L]" but the reference source code [and test vectors] say
66  otherwise.
67 */
68  L = (md->md2.chksum[j] ^= PI_SUBST[(int)(md->md2.buf[j] ^ L)] & 255);
69  }
70 }

72 static void md2_compress(hash_state *md)
73 {
74  int j, k;
75  unsigned char t;
76 
77  /* copy block */
78  for (j = 0; j < 16; j++) {
79  md->md2.X[16+j] = md->md2.buf[j];
80  md->md2.X[32+j] = md->md2.X[j] ^ md->md2.X[16+j];
81  }
82 
83  t = (unsigned char)0;
84 
85  /* do 18 rounds */
86  for (j = 0; j < 18; j++) {
87  for (k = 0; k < 48; k++) {
88  t = (md->md2.X[k] ^= PI_SUBST[(int)(t & 255)]);
89  }
90  t = (t + (unsigned char)j) & 255;
91  }

111 /**
112  Process a block of memory though the hash
113  @param md The hash state
114  @param in The data to hash
115  @param inlen The length of the data (octets)
116  @return CRYPT_OK if successful
117 */
118 int md2_process(hash_state *md, const unsigned char *in, unsigned long inlen)
119 {
120  unsigned long n;
121  LTC_ARGCHK(md != NULL);
122  LTC_ARGCHK(in != NULL);
123  if (md-> md2 .curlen > sizeof(md-> md2 .buf)) {
124  return CRYPT_INVALID_ARG;
125  }
126  while (inlen > 0) {
127  n = MIN(inlen, (16 - md->md2.curlen));
128  XMEMCPY(md->md2.buf + md->md2.curlen, in, (size_t)n);
129  md->md2.curlen += n;
130  in += n;
131  inlen -= n;
132 
133  /* is 16 bytes full? */
134  if (md->md2.curlen == 16) {
135  md2_compress(md);
136  md2_update_chksum(md);
137  md->md2.curlen = 0;
138  }
139  }
140  return CRYPT_OK;
141 }

147  @return CRYPT_OK if successful

149 int md2_done(hash_state * md, unsigned char *out)
150 {

152 
153  LTC_ARGCHK(md != NULL);
154  LTC_ARGCHK(out != NULL);
155 
156  if (md->md2.curlen >= sizeof(md->md2.buf)) {
157  return CRYPT_INVALID_ARG;
158  }
159 
160 
161  /* pad the message */
162  k = 16 - md->md2.curlen;
163  for (i = md->md2.curlen; i < 16; i++) {
164  md->md2.buf[i] = (unsigned char)k;
165  }
166 
167  /* hash and update */
168  md2_compress(md);
169  md2_update_chksum(md);
170 
171  /* hash checksum */
172  XMEMCPY(md->md2.buf, md->md2.chksum, 16);
173  md2_compress(md);
174 
175  /* output is lower 16 bytes of X */
176  XMEMCPY(out, md->md2.X, 16);
