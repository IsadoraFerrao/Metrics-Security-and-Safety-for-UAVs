26 static void _salsa20_block(unsigned char *output, const ulong32 *input, int rounds)
27 {
28  ulong32 x[16];
29  int i;
30  XMEMCPY(x, input, sizeof(x));
31  for (i = rounds; i > 0; i -= 2) {
32  QUARTERROUND( 0, 4, 8,12)
33  QUARTERROUND( 5, 9,13, 1)
34  QUARTERROUND(10,14, 2, 6)
35  QUARTERROUND(15, 3, 7,11)
36  QUARTERROUND( 0, 1, 2, 3)
37  QUARTERROUND( 5, 6, 7, 4)
38  QUARTERROUND(10,11, 8, 9)
39  QUARTERROUND(15,12,13,14)
40  }
41  for (i = 0; i < 16; ++i) {
42  x[i] += input[i];
43  STORE32L(x[i], output + 4 * i);
44  }
45 }

64  LTC_ARGCHK(out != NULL);
65  LTC_ARGCHK(st->ivlen == 8 || st->ivlen == 24);
66 
67  if (st->ksleft > 0) {
68  j = MIN(st->ksleft, inlen);
69  for (i = 0; i < j; ++i, st->ksleft--) out[i] = in[i] ^ st->kstream[64 - st->ksleft];
70  inlen -= j;
71  if (inlen == 0) return CRYPT_OK;
72  out += j;
73  in += j;
74  }
75  for (;;) {
76  _salsa20_block(buf, st->input, st->rounds);
77  /* Salsa20: 64-bit IV, increment 64-bit counter */
78  if (0 == ++st->input[8] && 0 == ++st->input[9]) return CRYPT_OVERFLOW;
79  if (inlen <= 64) {
80  for (i = 0; i < inlen; ++i) out[i] = in[i] ^ buf[i];
81  st->ksleft = 64 - inlen;
82  for (i = inlen; i < 64; ++i) st->kstream[i] = buf[i];
83  return CRYPT_OK;
84  }
85  for (i = 0; i < 64; ++i) out[i] = in[i] ^ buf[i];
