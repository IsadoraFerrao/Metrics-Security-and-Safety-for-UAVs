184 #define G(x) (SS3[((x)>>24)&255] ^ SS2[((x)>>16)&255] ^ SS1[((x)>>8)&255] ^ SS0[(x)&255])

186 #define F(L1, L2, R1, R2, K1, K2) \
187  T2 = G((R1 ^ K1) ^ (R2 ^ K2)); \
188  T = G( G(T2 + (R1 ^ K1)) + T2); \
189  L2 ^= T; \
190  L1 ^= (T + G(T2 + (R1 ^ K1))); \

192  /**
193  Initialize the SEED block cipher
194  @param key The symmetric key you wish to pass
195  @param keylen The key length in bytes
196  @param num_rounds The number of rounds desired (0 for default)
197  @param skey The key in as scheduled by this function.
198  @return CRYPT_OK if successful
199  */
200 int kseed_setup(const unsigned char *key, int keylen, int num_rounds, symmetric_key *skey)
201 {
202  int i;
203  ulong32 tmp, k1, k2, k3, k4;
204 
205  if (keylen != 16) {
206  return CRYPT_INVALID_KEYSIZE;
207  }
208 
209  if (num_rounds != 16 && num_rounds != 0) {
210  return CRYPT_INVALID_ROUNDS;
211  }
212 
213  /* load key */
214  LOAD32H(k1, key);
215  LOAD32H(k2, key+4);
216  LOAD32H(k3, key+8);
217  LOAD32H(k4, key+12);
218 
219  for (i = 0; i < 16; i++) {
220  skey->kseed.K[2*i+0] = G(k1 + k3 - KCi[i]);
221  skey->kseed.K[2*i+1] = G(k2 - k4 + KCi[i]);
222  if (i&1) {
223  tmp = k3;
224  k3 = ((k3 << 8) | (k4 >> 24)) & 0xFFFFFFFF;
225  k4 = ((k4 << 8) | (tmp >> 24)) & 0xFFFFFFFF;
226  } else {
227  tmp = k1;
228  k1 = ((k1 >> 8) | (k2 << 24)) & 0xFFFFFFFF;
229  k2 = ((k2 >> 8) | (tmp << 24)) & 0xFFFFFFFF;
230  }
231  /* reverse keys for decrypt */
232  skey->kseed.dK[2*(15-i)+0] = skey->kseed.K[2*i+0];
233  skey->kseed.dK[2*(15-i)+1] = skey->kseed.K[2*i+1];
234  }
235 
236  return CRYPT_OK;
237 }
