53 /*
54  * Serpent S-boxes, implemented in bitslice mode. These circuits have
55  * been published by Dag Arne Osvik ("Speeding up Serpent", published in
56  * the 3rd AES Candidate Conference) and work on five 32-bit registers:
57  * the four inputs, and a fifth scratch register. There are meant to be
58  * quite fast on Pentium-class processors. These are not the fastest
59  * published, but they are "fast enough" and they are unencumbered as
60  * far as intellectual property is concerned (note: these are rewritten
61  * from the article itself, and hence are not covered by the GPL on
62  * Dag's code, which was not used here).
63  *
64  * The output bits are permuted. Here is the correspondance:
65  * S0: 1420
66  * S1: 2031
67  * S2: 2314
68  * S3: 1234
69  * S4: 1403
70  * S5: 1302
71  * S6: 0142
72  * S7: 4310
73  * (for instance, the output of S0 is in "r1, r4, r2, r0").
74  */
75 
76 #define S0(r0, r1, r2, r3, r4) do { \
77  r3 ^= r0; r4 = r1; \
78  r1 &= r3; r4 ^= r2; \
79  r1 ^= r0; r0 |= r3; \
80  r0 ^= r4; r4 ^= r3; \
81  r3 ^= r2; r2 |= r1; \
82  r2 ^= r4; r4 = ~r4; \
83  r4 |= r1; r1 ^= r3; \
84  r1 ^= r4; r3 |= r0; \
85  r1 ^= r3; r4 ^= r3; \
86  } while (0)
87 
88 #define S1(r0, r1, r2, r3, r4) do { \
89  r0 = ~r0; r2 = ~r2; \
90  r4 = r0; r0 &= r1; \
91  r2 ^= r0; r0 |= r3; \
92  r3 ^= r2; r1 ^= r0; \
93  r0 ^= r4; r4 |= r1; \
94  r1 ^= r3; r2 |= r0; \
95  r2 &= r4; r0 ^= r1; \
96  r1 &= r2; \
97  r1 ^= r0; r0 &= r2; \
98  r0 ^= r4; \
99  } while (0)
100 
101 #define S2(r0, r1, r2, r3, r4) do { \
102  r4 = r0; r0 &= r2; \
103  r0 ^= r3; r2 ^= r1; \
104  r2 ^= r0; r3 |= r4; \
105  r3 ^= r1; r4 ^= r2; \
106  r1 = r3; r3 |= r4; \
107  r3 ^= r0; r0 &= r1; \
108  r4 ^= r0; r1 ^= r3; \
109  r1 ^= r4; r4 = ~r4; \
110  } while (0)
111 
112 #define S3(r0, r1, r2, r3, r4) do { \
113  r4 = r0; r0 |= r3; \
114  r3 ^= r1; r1 &= r4; \
115  r4 ^= r2; r2 ^= r3; \
116  r3 &= r0; r4 |= r1; \
117  r3 ^= r4; r0 ^= r1; \
118  r4 &= r0; r1 ^= r3; \
119  r4 ^= r2; r1 |= r0; \
120  r1 ^= r2; r0 ^= r3; \
121  r2 = r1; r1 |= r3; \
122  r1 ^= r0; \
123  } while (0)
124 
125 #define S4(r0, r1, r2, r3, r4) do { \
126  r1 ^= r3; r3 = ~r3; \
127  r2 ^= r3; r3 ^= r0; \
128  r4 = r1; r1 &= r3; \
129  r1 ^= r2; r4 ^= r3; \
130  r0 ^= r4; r2 &= r4; \
131  r2 ^= r0; r0 &= r1; \
132  r3 ^= r0; r4 |= r1; \
133  r4 ^= r0; r0 |= r3; \
134  r0 ^= r2; r2 &= r3; \
135  r0 = ~r0; r4 ^= r2; \
136  } while (0)
137 
138 #define S5(r0, r1, r2, r3, r4) do { \
139  r0 ^= r1; r1 ^= r3; \
140  r3 = ~r3; r4 = r1; \
141  r1 &= r0; r2 ^= r3; \
142  r1 ^= r2; r2 |= r4; \
143  r4 ^= r3; r3 &= r1; \
144  r3 ^= r0; r4 ^= r1; \
145  r4 ^= r2; r2 ^= r0; \
146  r0 &= r3; r2 = ~r2; \
147  r0 ^= r4; r4 |= r3; \
148  r2 ^= r4; \
149  } while (0)
150 
151 #define S6(r0, r1, r2, r3, r4) do { \
152  r2 = ~r2; r4 = r3; \
153  r3 &= r0; r0 ^= r4; \
154  r3 ^= r2; r2 |= r4; \
155  r1 ^= r3; r2 ^= r0; \
156  r0 |= r1; r2 ^= r1; \
157  r4 ^= r0; r0 |= r3; \
158  r0 ^= r2; r4 ^= r3; \
159  r4 ^= r0; r3 = ~r3; \
160  r2 &= r4; \
161  r2 ^= r3; \
162  } while (0)
163 
164 #define S7(r0, r1, r2, r3, r4) do { \
165  r4 = r1; r1 |= r2; \
166  r1 ^= r3; r4 ^= r2; \
167  r2 ^= r1; r3 |= r4; \
168  r3 &= r0; r4 ^= r2; \
169  r3 ^= r1; r1 |= r4; \
170  r1 ^= r0; r0 |= r4; \
171  r0 ^= r2; r1 ^= r4; \
172  r2 ^= r1; r1 &= r0; \
173  r1 ^= r4; r2 = ~r2; \
174  r2 |= r0; \
175  r4 ^= r2; \
176  } while (0)

178 /*
179  * The Serpent linear transform.
180  */
181 #define SERPENT_LT(x0, x1, x2, x3) do { \
182  x0 = ROLc(x0, 13); \
183  x2 = ROLc(x2, 3); \
184  x1 = x1 ^ x0 ^ x2; \
185  x3 = x3 ^ x2 ^ T32(x0 << 3); \
186  x1 = ROLc(x1, 1); \
187  x3 = ROLc(x3, 7); \
188  x0 = x0 ^ x1 ^ x3; \
189  x2 = x2 ^ x3 ^ T32(x1 << 7); \
190  x0 = ROLc(x0, 5); \
191  x2 = ROLc(x2, 22); \
192  } while (0)

196 /*
197  * Initialize Sosemanuk's state by providing a key. The key is an array of
198  * 1 to 32 bytes.
199  * @param st The Sosemanuk state
200  * @param key Key
201  * @param keylen Length of key in bytes
202  * @return CRYPT_OK on success
203  */
204 int sosemanuk_setup(sosemanuk_state *st, const unsigned char *key, unsigned long keylen)
205 {
206  /*
207  * This key schedule is actually a truncated Serpent key schedule.
208  * The key-derived words (w_i) are computed within the eight
209  * local variables w0 to w7, which are reused again and again.
210  */
211 
212 #define SKS(S, o0, o1, o2, o3, d0, d1, d2, d3) do { \
213  ulong32 r0, r1, r2, r3, r4; \
214  r0 = w ## o0; \
215  r1 = w ## o1; \
216  r2 = w ## o2; \
217  r3 = w ## o3; \
218  S(r0, r1, r2, r3, r4); \
219  st->kc[i ++] = r ## d0; \
220  st->kc[i ++] = r ## d1; \
221  st->kc[i ++] = r ## d2; \
222  st->kc[i ++] = r ## d3; \
223  } while (0)
224 
225 #define SKS0 SKS(S0, 4, 5, 6, 7, 1, 4, 2, 0)
226 #define SKS1 SKS(S1, 0, 1, 2, 3, 2, 0, 3, 1)
227 #define SKS2 SKS(S2, 4, 5, 6, 7, 2, 3, 1, 4)
228 #define SKS3 SKS(S3, 0, 1, 2, 3, 1, 2, 3, 4)
229 #define SKS4 SKS(S4, 4, 5, 6, 7, 1, 4, 0, 3)
230 #define SKS5 SKS(S5, 0, 1, 2, 3, 1, 3, 0, 2)
231 #define SKS6 SKS(S6, 4, 5, 6, 7, 0, 1, 4, 2)
232 #define SKS7 SKS(S7, 0, 1, 2, 3, 4, 3, 1, 0)
233 
234 #define WUP(wi, wi5, wi3, wi1, cc) do { \
235  ulong32 tt = (wi) ^ (wi5) ^ (wi3) \
236  ^ (wi1) ^ (0x9E3779B9 ^ (ulong32)(cc)); \
237  (wi) = ROLc(tt, 11); \
238  } while (0)
239 
240 #define WUP0(cc) do { \
241  WUP(w0, w3, w5, w7, cc); \
242  WUP(w1, w4, w6, w0, cc + 1); \
243  WUP(w2, w5, w7, w1, cc + 2); \
244  WUP(w3, w6, w0, w2, cc + 3); \
245  } while (0)
246 
247 #define WUP1(cc) do { \
248  WUP(w4, w7, w1, w3, cc); \
249  WUP(w5, w0, w2, w4, cc + 1); \
250  WUP(w6, w1, w3, w5, cc + 2); \
251  WUP(w7, w2, w4, w6, cc + 3); \
252  } while (0)
253 
254  unsigned char wbuf[32];
255  ulong32 w0, w1, w2, w3, w4, w5, w6, w7;
256  int i = 0;
257 
258  LTC_ARGCHK(st != NULL);
259  LTC_ARGCHK(key != NULL);
260  LTC_ARGCHK(keylen > 0 && keylen <= 32);
261 
262  /*
263  * The key is copied into the wbuf[] buffer and padded to 256 bits
264  * as described in the Serpent specification.
265  */
266  XMEMCPY(wbuf, key, keylen);
267  if (keylen < 32) {
268  wbuf[keylen] = 0x01;
269  if (keylen < 31) {
270  XMEMSET(wbuf + keylen + 1, 0, 31 - keylen);
271  }
272  }
273 
274  LOAD32L(w0, wbuf);
275  LOAD32L(w1, wbuf + 4);
276  LOAD32L(w2, wbuf + 8);
277  LOAD32L(w3, wbuf + 12);
278  LOAD32L(w4, wbuf + 16);
279  LOAD32L(w5, wbuf + 20);
280  LOAD32L(w6, wbuf + 24);
281  LOAD32L(w7, wbuf + 28);
282 
283  WUP0(0); SKS3;
284  WUP1(4); SKS2;
285  WUP0(8); SKS1;
286  WUP1(12); SKS0;
287  WUP0(16); SKS7;
288  WUP1(20); SKS6;
289  WUP0(24); SKS5;
290  WUP1(28); SKS4;
291  WUP0(32); SKS3;
292  WUP1(36); SKS2;
293  WUP0(40); SKS1;
294  WUP1(44); SKS0;
295  WUP0(48); SKS7;
296  WUP1(52); SKS6;
297  WUP0(56); SKS5;
298  WUP1(60); SKS4;
299  WUP0(64); SKS3;
300  WUP1(68); SKS2;
301  WUP0(72); SKS1;
302  WUP1(76); SKS0;
303  WUP0(80); SKS7;
304  WUP1(84); SKS6;
305  WUP0(88); SKS5;
306  WUP1(92); SKS4;
307  WUP0(96); SKS3;
308 
309 #undef SKS
310 #undef SKS0
311 #undef SKS1
312 #undef SKS2
313 #undef SKS3
314 #undef SKS4
315 #undef SKS5
316 #undef SKS6
317 #undef SKS7
318 #undef WUP
319 #undef WUP0
320 #undef WUP1
321 
322  return CRYPT_OK;
323 }

326 /*
327  * Initialization continues by setting the IV. The IV length is up to 16 bytes.
328  * If "ivlen" is 0 (no IV), then the "iv" parameter can be NULL. If multiple
329  * encryptions/decryptions are to be performed with the same key and
330  * sosemanuk_done() has not been called, only sosemanuk_setiv() need be called
331  * to set the state.
332  * @param st The Sosemanuk state
333  * @param iv Initialization vector
334  * @param ivlen Length of iv in bytes
335  * @return CRYPT_OK on success
336  */
337 int sosemanuk_setiv(sosemanuk_state *st, const unsigned char *iv, unsigned long ivlen)
338 {
339 
340  /*
341  * The Serpent key addition step.
342  */
343 #define KA(zc, x0, x1, x2, x3) do { \
344  x0 ^= st->kc[(zc)]; \
345  x1 ^= st->kc[(zc) + 1]; \
346  x2 ^= st->kc[(zc) + 2]; \
347  x3 ^= st->kc[(zc) + 3]; \
348  } while (0)
349 
350  /*
351  * One Serpent round.
352  * zc = current subkey counter
353  * S = S-box macro for this round
354  * i0 to i4 = input register numbers (the fifth is a scratch register)
355  * o0 to o3 = output register numbers
356  */
357 #define FSS(zc, S, i0, i1, i2, i3, i4, o0, o1, o2, o3) do { \
358  KA(zc, r ## i0, r ## i1, r ## i2, r ## i3); \
359  S(r ## i0, r ## i1, r ## i2, r ## i3, r ## i4); \
360  SERPENT_LT(r ## o0, r ## o1, r ## o2, r ## o3); \
361  } while (0)
362 
363  /*
364  * Last Serpent round. Contrary to the "true" Serpent, we keep
365  * the linear transformation for that last round.
366  */
367 #define FSF(zc, S, i0, i1, i2, i3, i4, o0, o1, o2, o3) do { \
368  KA(zc, r ## i0, r ## i1, r ## i2, r ## i3); \
369  S(r ## i0, r ## i1, r ## i2, r ## i3, r ## i4); \
370  SERPENT_LT(r ## o0, r ## o1, r ## o2, r ## o3); \
371  KA(zc + 4, r ## o0, r ## o1, r ## o2, r ## o3); \
372  } while (0)
373 
374  ulong32 r0, r1, r2, r3, r4;
375  unsigned char ivtmp[16] = {0};
376 
377  LTC_ARGCHK(st != NULL);
378  LTC_ARGCHK(ivlen <= 16);
379  LTC_ARGCHK(iv != NULL || ivlen == 0);
380 
381  if (ivlen > 0) XMEMCPY(ivtmp, iv, ivlen);
382 
383  /*
384  * Decode IV into four 32-bit words (little-endian).
385  */
386  LOAD32L(r0, ivtmp);
387  LOAD32L(r1, ivtmp + 4);
388  LOAD32L(r2, ivtmp + 8);
389  LOAD32L(r3, ivtmp + 12);
390 
391  /*
392  * Encrypt IV with Serpent24. Some values are extracted from the
393  * output of the twelfth, eighteenth and twenty-fourth rounds.
394  */
395  FSS(0, S0, 0, 1, 2, 3, 4, 1, 4, 2, 0);
396  FSS(4, S1, 1, 4, 2, 0, 3, 2, 1, 0, 4);
397  FSS(8, S2, 2, 1, 0, 4, 3, 0, 4, 1, 3);
398  FSS(12, S3, 0, 4, 1, 3, 2, 4, 1, 3, 2);
399  FSS(16, S4, 4, 1, 3, 2, 0, 1, 0, 4, 2);
400  FSS(20, S5, 1, 0, 4, 2, 3, 0, 2, 1, 4);
401  FSS(24, S6, 0, 2, 1, 4, 3, 0, 2, 3, 1);
402  FSS(28, S7, 0, 2, 3, 1, 4, 4, 1, 2, 0);
403  FSS(32, S0, 4, 1, 2, 0, 3, 1, 3, 2, 4);
404  FSS(36, S1, 1, 3, 2, 4, 0, 2, 1, 4, 3);
405  FSS(40, S2, 2, 1, 4, 3, 0, 4, 3, 1, 0);
406  FSS(44, S3, 4, 3, 1, 0, 2, 3, 1, 0, 2);
407  st->s09 = r3;
408  st->s08 = r1;
409  st->s07 = r0;
410  st->s06 = r2;
411 
412  FSS(48, S4, 3, 1, 0, 2, 4, 1, 4, 3, 2);
413  FSS(52, S5, 1, 4, 3, 2, 0, 4, 2, 1, 3);
414  FSS(56, S6, 4, 2, 1, 3, 0, 4, 2, 0, 1);
415  FSS(60, S7, 4, 2, 0, 1, 3, 3, 1, 2, 4);
416  FSS(64, S0, 3, 1, 2, 4, 0, 1, 0, 2, 3);
417  FSS(68, S1, 1, 0, 2, 3, 4, 2, 1, 3, 0);
418  st->r1 = r2;
419  st->s04 = r1;
420  st->r2 = r3;
421  st->s05 = r0;
422 
423  FSS(72, S2, 2, 1, 3, 0, 4, 3, 0, 1, 4);
424  FSS(76, S3, 3, 0, 1, 4, 2, 0, 1, 4, 2);
425  FSS(80, S4, 0, 1, 4, 2, 3, 1, 3, 0, 2);
426  FSS(84, S5, 1, 3, 0, 2, 4, 3, 2, 1, 0);
427  FSS(88, S6, 3, 2, 1, 0, 4, 3, 2, 4, 1);
428  FSF(92, S7, 3, 2, 4, 1, 0, 0, 1, 2, 3);
429  st->s03 = r0;
430  st->s02 = r1;
431  st->s01 = r2;
432  st->s00 = r3;
433 
434  st->ptr = sizeof(st->buf);
435 
436 #undef KA
437 #undef FSS
438 #undef FSF
439 
440  return CRYPT_OK;
441 }

584 /*
585  * Compute the next block of bits of output stream. This is equivalent
586  * to one full rotation of the shift register.
587  */
588 static LTC_INLINE void _sosemanuk_internal(sosemanuk_state *st)
589 {
590  /*
591  * MUL_A(x) computes alpha * x (in F_{2^32}).
592  * MUL_G(x) computes 1/alpha * x (in F_{2^32}).
593  */
594 #define MUL_A(x) (T32((x) << 8) ^ mul_a[(x) >> 24])
595 #define MUL_G(x) (((x) >> 8) ^ mul_ia[(x) & 0xFF])
596 
597  /*
598  * This macro computes the special multiplexer, which chooses
599  * between "x" and "x xor y", depending on the least significant
600  * bit of the control word. We use the C "?:" selection operator
601  * (which most compilers know how to optimise) except for Alpha,
602  * where the manual sign extension seems to perform equally well
603  * with DEC/Compaq/HP compiler, and much better with gcc.
604  */
605 #ifdef __alpha
606 #define XMUX(c, x, y) ((((signed int)((c) << 31) >> 31) & (y)) ^ (x))
607 #else
608 #define XMUX(c, x, y) (((c) & 0x1) ? ((x) ^ (y)) : (x))
609 #endif
610 
611  /*
612  * FSM() updates the finite state machine.
613  */
614 #define FSM(x0, x1, x2, x3, x4, x5, x6, x7, x8, x9) do { \
615  ulong32 tt, or1; \
616  tt = XMUX(r1, s ## x1, s ## x8); \
617  or1 = r1; \
618  r1 = T32(r2 + tt); \
619  tt = T32(or1 * 0x54655307); \
620  r2 = ROLc(tt, 7); \
621  } while (0)
622 
623  /*
624  * LRU updates the shift register; the dropped value is stored
625  * in variable "dd".
626  */
627 #define LRU(x0, x1, x2, x3, x4, x5, x6, x7, x8, x9, dd) do { \
628  dd = s ## x0; \
629  s ## x0 = MUL_A(s ## x0) ^ MUL_G(s ## x3) ^ s ## x9; \
630  } while (0)
631 
632  /*
633  * CC1 stores into variable "ee" the next intermediate word
634  * (combination of the new states of the LFSR and the FSM).
635  */
636 #define CC1(x0, x1, x2, x3, x4, x5, x6, x7, x8, x9, ee) do { \
637  ee = T32(s ## x9 + r1) ^ r2; \
638  } while (0)
639 
640  /*
641  * STEP computes one internal round. "dd" receives the "s_t"
642  * value (dropped from the LFSR) and "ee" gets the value computed
643  * from the LFSR and FSM.
644  */
645 #define STEP(x0, x1, x2, x3, x4, x5, x6, x7, x8, x9, dd, ee) do { \
646  FSM(x0, x1, x2, x3, x4, x5, x6, x7, x8, x9); \
647  LRU(x0, x1, x2, x3, x4, x5, x6, x7, x8, x9, dd); \
648  CC1(x0, x1, x2, x3, x4, x5, x6, x7, x8, x9, ee); \
649  } while (0)
650

652  * Apply one Serpent round (with the provided S-box macro), XOR
653  * the result with the "v" values, and encode the result into
654  * the destination buffer, at the provided offset. The "x*"
655  * arguments encode the output permutation of the "S" macro.
656  */
657 #define SRD(S, x0, x1, x2, x3, ooff) do { \
658  S(u0, u1, u2, u3, u4); \
659  STORE32L(u ## x0 ^ v0, st->buf + ooff); \
660  STORE32L(u ## x1 ^ v1, st->buf + ooff + 4); \
661  STORE32L(u ## x2 ^ v2, st->buf + ooff + 8); \
662  STORE32L(u ## x3 ^ v3, st->buf + ooff + 12); \
663  } while (0)

666  ulong32 s01 = st->s01;
667  ulong32 s02 = st->s02;
668  ulong32 s03 = st->s03;

702  STEP(08, 09, 00, 01, 02, 03, 04, 05, 06, 07, v2, u2);
703  STEP(09, 00, 01, 02, 03, 04, 05, 06, 07, 08, v3, u3);

720 /*
721  * Combine buffers in1[] and in2[] by XOR, result in out[]. The length
722  * is "datalen" (in bytes). Partial overlap of out[] with either in1[]
723  * or in2[] is not allowed. Total overlap (out == in1 and/or out == in2)
724  * is allowed.
725  */
726 static LTC_INLINE void _xorbuf(const unsigned char *in1, const unsigned char *in2,
727  unsigned char *out, unsigned long datalen)
728 {
729  while (datalen -- > 0) {
730  *out ++ = *in1 ++ ^ *in2 ++;
731  }
732 }

735 /*
736  * Cipher operation, as a stream cipher: data is read from the "in"
737  * buffer, combined by XOR with the stream, and the result is written
738  * in the "out" buffer. "in" and "out" must be either equal, or
739  * reference distinct buffers (no partial overlap is allowed).
740  * @param st The Sosemanuk state
741  * @param in Data in
742  * @param inlen Length of data in bytes
743  * @param out Data out
744  * @return CRYPT_OK on success
745  */
746 int sosemanuk_crypt(sosemanuk_state *st,
747  const unsigned char *in, unsigned long inlen, unsigned char *out)
748 {
749  LTC_ARGCHK(st != NULL);
750  LTC_ARGCHK(in != NULL);
751  LTC_ARGCHK(out != NULL);
752 
753  if (st->ptr < (sizeof(st->buf))) {
754  unsigned long rlen = (sizeof(st->buf)) - st->ptr;
755 
756  if (rlen > inlen) {
757  rlen = inlen;
758  }
759  _xorbuf(st->buf + st->ptr, in, out, rlen);
760  in += rlen;
761  out += rlen;
762  inlen -= rlen;
763  st->ptr += rlen;
764  }
765  while (inlen > 0) {
766  _sosemanuk_internal(st);
767  if (inlen >= sizeof(st->buf)) {
768  _xorbuf(st->buf, in, out, sizeof(st->buf));
769  in += sizeof(st->buf);
770  out += sizeof(st->buf);
771  inlen -= sizeof(st->buf);
772  } else {
773  _xorbuf(st->buf, in, out, inlen);
774  st->ptr = inlen;
775  inlen = 0;
776  }
777  }
778  return CRYPT_OK;
779 }
