944  (T5[(K0 >> 16) & 0xff] & 0x00ff0000U) ^
945  (T5[(K0 >> 8) & 0xff] & 0x0000ff00U) ^
946  (T5[(K0 ) & 0xff] & 0x000000ffU);
947  K1 = T4[(kappa[i] >> 16) & 0xff] ^
948  (T5[(K1 >> 24) & 0xff] & 0xff000000U) ^
949  (T5[(K1 >> 16) & 0xff] & 0x00ff0000U) ^
950  (T5[(K1 >> 8) & 0xff] & 0x0000ff00U) ^
951  (T5[(K1 ) & 0xff] & 0x000000ffU);
952  K2 = T4[(kappa[i] >> 8) & 0xff] ^
953  (T5[(K2 >> 24) & 0xff] & 0xff000000U) ^
954  (T5[(K2 >> 16) & 0xff] & 0x00ff0000U) ^
955  (T5[(K2 >> 8) & 0xff] & 0x0000ff00U) ^
956  (T5[(K2 ) & 0xff] & 0x000000ffU);
957  K3 = T4[(kappa[i] ) & 0xff] ^
958  (T5[(K3 >> 24) & 0xff] & 0xff000000U) ^
959  (T5[(K3 >> 16) & 0xff] & 0x00ff0000U) ^
960  (T5[(K3 >> 8) & 0xff] & 0x0000ff00U) ^
961  (T5[(K3 ) & 0xff] & 0x000000ffU);
962  }
963  /*
964  -- this is the code to use with the large U tables:
965  K0 = K1 = K2 = K3 = 0;
966  for (i = 0; i < N; i++) {
967  K0 ^= U[i][(kappa[i] >> 24) & 0xff];
968  K1 ^= U[i][(kappa[i] >> 16) & 0xff];
969  K2 ^= U[i][(kappa[i] >> 8) & 0xff];
970  K3 ^= U[i][(kappa[i] ) & 0xff];
971  }
972  */
973  skey->anubis.roundKeyEnc[r][0] = K0;
974  skey->anubis.roundKeyEnc[r][1] = K1;
975  skey->anubis.roundKeyEnc[r][2] = K2;
976  skey->anubis.roundKeyEnc[r][3] = K3;
977 
978  /*
979  * compute kappa^{r+1} from kappa^r:
980  */
981  if (r == R) {
982  break;
983  }
984  for (i = 0; i < N; i++) {
985  int j = i;
986  inter[i] = T0[(kappa[j--] >> 24) & 0xff]; if (j < 0) j = N - 1;
987  inter[i] ^= T1[(kappa[j--] >> 16) & 0xff]; if (j < 0) j = N - 1;
988  inter[i] ^= T2[(kappa[j--] >> 8) & 0xff]; if (j < 0) j = N - 1;
989  inter[i] ^= T3[(kappa[j ] ) & 0xff];
990  }
991  kappa[0] = inter[0] ^ rc[r];
992  for (i = 1; i < N; i++) {
993  kappa[i] = inter[i];
994  }
995  }
996 
997  /*
998  * generate inverse key schedule: K'^0 = K^R, K'^R = K^0, K'^r = theta(K^{R-r}):
999  */
1000  for (i = 0; i < 4; i++) {
1001  skey->anubis.roundKeyDec[0][i] = skey->anubis.roundKeyEnc[R][i];
1002  skey->anubis.roundKeyDec[R][i] = skey->anubis.roundKeyEnc[0][i];
1003  }
1004  for (r = 1; r < R; r++) {
1005  for (i = 0; i < 4; i++) {
1006  v = skey->anubis.roundKeyEnc[R - r][i];
1007  skey->anubis.roundKeyDec[r][i] =
1008  T0[T4[(v >> 24) & 0xff] & 0xff] ^
1009  T1[T4[(v >> 16) & 0xff] & 0xff] ^
1010  T2[T4[(v >> 8) & 0xff] & 0xff] ^
1011  T3[T4[(v ) & 0xff] & 0xff];
1012  }
1013  }
1014 
1015  return CRYPT_OK;
1016 }

1029 static void anubis_crypt(const unsigned char *plaintext, unsigned char *ciphertext,
1030  const ulong32 roundKey[18 + 1][4], int R) {
1031  int i, pos, r;
1032  ulong32 state[4];
1033  ulong32 inter[4];
1034 
1035  /*
1036  * map plaintext block to cipher state (mu)
1037  * and add initial round key (sigma[K^0]):
1038  */
1039  for (i = 0, pos = 0; i < 4; i++, pos += 4) {
1040  state[i] =
1041  (((ulong32)plaintext[pos ]) << 24) ^
1042  (((ulong32)plaintext[pos + 1]) << 16) ^
1043  (((ulong32)plaintext[pos + 2]) << 8) ^
1044  (((ulong32)plaintext[pos + 3]) ) ^
1045  roundKey[0][i];
1046  }
1047 
1048  /*
1049  * R - 1 full rounds:
1050  */
1051  for (r = 1; r < R; r++) {
1052  inter[0] =
1053  T0[(state[0] >> 24) & 0xff] ^
1054  T1[(state[1] >> 24) & 0xff] ^
1055  T2[(state[2] >> 24) & 0xff] ^
1056  T3[(state[3] >> 24) & 0xff] ^
1057  roundKey[r][0];
1058  inter[1] =
1059  T0[(state[0] >> 16) & 0xff] ^
1060  T1[(state[1] >> 16) & 0xff] ^
1061  T2[(state[2] >> 16) & 0xff] ^
1062  T3[(state[3] >> 16) & 0xff] ^
1063  roundKey[r][1];
1064  inter[2] =
1065  T0[(state[0] >> 8) & 0xff] ^
1066  T1[(state[1] >> 8) & 0xff] ^
1067  T2[(state[2] >> 8) & 0xff] ^
1068  T3[(state[3] >> 8) & 0xff] ^
1069  roundKey[r][2];
1070  inter[3] =
1071  T0[(state[0] ) & 0xff] ^
1072  T1[(state[1] ) & 0xff] ^
1073  T2[(state[2] ) & 0xff] ^
1074  T3[(state[3] ) & 0xff] ^
1075  roundKey[r][3];
1076  state[0] = inter[0];
1077  state[1] = inter[1];
1078  state[2] = inter[2];
1079  state[3] = inter[3];
1080  }
1081 
1082  /*
1083  * last round:
1084  */
1085  inter[0] =
1086  (T0[(state[0] >> 24) & 0xff] & 0xff000000U) ^
1087  (T1[(state[1] >> 24) & 0xff] & 0x00ff0000U) ^
1088  (T2[(state[2] >> 24) & 0xff] & 0x0000ff00U) ^
1089  (T3[(state[3] >> 24) & 0xff] & 0x000000ffU) ^
1090  roundKey[R][0];
1091  inter[1] =
1092  (T0[(state[0] >> 16) & 0xff] & 0xff000000U) ^
1093  (T1[(state[1] >> 16) & 0xff] & 0x00ff0000U) ^
1094  (T2[(state[2] >> 16) & 0xff] & 0x0000ff00U) ^
1095  (T3[(state[3] >> 16) & 0xff] & 0x000000ffU) ^
1096  roundKey[R][1];
1097  inter[2] =
1098  (T0[(state[0] >> 8) & 0xff] & 0xff000000U) ^
1099  (T1[(state[1] >> 8) & 0xff] & 0x00ff0000U) ^
1100  (T2[(state[2] >> 8) & 0xff] & 0x0000ff00U) ^
1101  (T3[(state[3] >> 8) & 0xff] & 0x000000ffU) ^
1102  roundKey[R][2];
1103  inter[3] =
1104  (T0[(state[0] ) & 0xff] & 0xff000000U) ^
1105  (T1[(state[1] ) & 0xff] & 0x00ff0000U) ^
1106  (T2[(state[2] ) & 0xff] & 0x0000ff00U) ^
1107  (T3[(state[3] ) & 0xff] & 0x000000ffU) ^
1108  roundKey[R][3];
1109 
1110  /*
1111  * map cipher state to ciphertext block (mu^{-1}):
1112  */
1113  for (i = 0, pos = 0; i < 4; i++, pos += 4) {
1114  ulong32 w = inter[i];
1115  ciphertext[pos ] = (unsigned char)(w >> 24);
1116  ciphertext[pos + 1] = (unsigned char)(w >> 16);
1117  ciphertext[pos + 2] = (unsigned char)(w >> 8);
1118  ciphertext[pos + 3] = (unsigned char)(w );
1119  }
1120 }
