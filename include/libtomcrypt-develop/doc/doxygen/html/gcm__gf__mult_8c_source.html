109 /**
110  GCM GF multiplier (internal use only) word oriented
111  @param a First value
112  @param b Second value
113  @param c Destination for a * b
114  */
115 void gcm_gf_mult(const unsigned char *a, const unsigned char *b, unsigned char *c)
116 {
117  int i, j, k, u;
118  LTC_FAST_TYPE B[16][WPV], tmp[32 / sizeof(LTC_FAST_TYPE)], pB[16 / sizeof(LTC_FAST_TYPE)], zz, z;
119  unsigned char pTmp[32];
120 
121  /* create simple tables */
122  zeromem(B[0], sizeof(B[0]));
123  zeromem(B[M(1)], sizeof(B[M(1)]));
124 
125 #ifdef ENDIAN_32BITWORD
126  for (i = 0; i < 4; i++) {
127  LOAD32H(B[M(1)][i], a + (i<<2));
128  LOAD32L(pB[i], b + (i<<2));
129  }
130 #else
131  for (i = 0; i < 2; i++) {
132  LOAD64H(B[M(1)][i], a + (i<<3));
133  LOAD64L(pB[i], b + (i<<3));
134  }
135 #endif
136 
137  /* now create 2, 4 and 8 */
138  B[M(2)][0] = B[M(1)][0] >> 1;
139  B[M(4)][0] = B[M(1)][0] >> 2;
140  B[M(8)][0] = B[M(1)][0] >> 3;
141  for (i = 1; i < (int)WPV; i++) {
142  B[M(2)][i] = (B[M(1)][i-1] << (BPD-1)) | (B[M(1)][i] >> 1);
143  B[M(4)][i] = (B[M(1)][i-1] << (BPD-2)) | (B[M(1)][i] >> 2);
144  B[M(8)][i] = (B[M(1)][i-1] << (BPD-3)) | (B[M(1)][i] >> 3);
145  }
146 
147  /* now all values with two bits which are 3, 5, 6, 9, 10, 12 */
148  for (i = 0; i < (int)WPV; i++) {
149  B[M(3)][i] = B[M(1)][i] ^ B[M(2)][i];
150  B[M(5)][i] = B[M(1)][i] ^ B[M(4)][i];
151  B[M(6)][i] = B[M(2)][i] ^ B[M(4)][i];
152  B[M(9)][i] = B[M(1)][i] ^ B[M(8)][i];
153  B[M(10)][i] = B[M(2)][i] ^ B[M(8)][i];
154  B[M(12)][i] = B[M(8)][i] ^ B[M(4)][i];
155 
156  /* now all 3 bit values and the only 4 bit value: 7, 11, 13, 14, 15 */
157  B[M(7)][i] = B[M(3)][i] ^ B[M(4)][i];
158  B[M(11)][i] = B[M(3)][i] ^ B[M(8)][i];
159  B[M(13)][i] = B[M(1)][i] ^ B[M(12)][i];
160  B[M(14)][i] = B[M(6)][i] ^ B[M(8)][i];
161  B[M(15)][i] = B[M(7)][i] ^ B[M(8)][i];
162  }
163

165 
166  /* compute product four bits of each word at a time */
167  /* for each nibble */
168  for (i = (BPD/4)-1; i >= 0; i--) {
169  /* for each word */
170  for (j = 0; j < (int)(WPV-1); j++) {
171  /* grab the 4 bits recall the nibbles are backwards so it's a shift by (i^1)*4 */
172  u = (pB[j] >> ((i^1)<<2)) & 15;
173 
174  /* add offset by the word count the table looked up value to the result */
175  for (k = 0; k < (int)WPV; k++) {
176  tmp[k+j] ^= B[u][k];
177  }
178  }
179  /* shift result up by 4 bits */
180  if (i != 0) {
181  for (z = j = 0; j < (int)(32 / sizeof(LTC_FAST_TYPE)); j++) {
182  zz = tmp[j] << (BPD-4);
183  tmp[j] = (tmp[j] >> 4) | z;
184  z = zz;
185  }
186  }
187  }
188 
189  /* store product */
190 #ifdef ENDIAN_32BITWORD
191  for (i = 0; i < 8; i++) {
192  STORE32H(tmp[i], pTmp + (i<<2));
193  }
194 #else
195  for (i = 0; i < 4; i++) {
196  STORE64H(tmp[i], pTmp + (i<<3));
197  }
198 #endif
199 
200  /* reduce by taking most significant byte and adding the appropriate two byte sequence 16 bytes down */
201  for (i = 31; i >= 16; i--) {
202  pTmp[i-16] ^= gcm_shift_table[((unsigned)pTmp[i]<<1)];
203  pTmp[i-15] ^= gcm_shift_table[((unsigned)pTmp[i]<<1)+1];
204  }
