77 {

79 #if defined(_WIN32)

101 /* reseed the PRNG */
102 static int _fortuna_reseed(prng_state *prng)
103 {
104  unsigned char tmp[MAXBLOCKSIZE];
105  hash_state md;
106  ulong64 reset_cnt;
107  int err, x;
108 
109 #ifdef LTC_FORTUNA_RESEED_RATELIMIT_TIMED
110  ulong64 now = _fortuna_current_time();
111  if (now == prng->u.fortuna.wd) {
112  return CRYPT_OK;
113  }
114 #else
115  if (++prng->u.fortuna.wd < LTC_FORTUNA_WD) {
116  return CRYPT_OK;
117  }
118 #endif
119 
120  /* new K == LTC_SHA256(K || s) where s == LTC_SHA256(P0) || LTC_SHA256(P1) ... */
121  sha256_init(&md);
122  if ((err = sha256_process(&md, prng->u.fortuna.K, 32)) != CRYPT_OK) {
123  sha256_done(&md, tmp);
124  return err;
125  }
126 
127  reset_cnt = prng->u.fortuna.reset_cnt + 1;
128 
129  for (x = 0; x < LTC_FORTUNA_POOLS; x++) {
130  if (x == 0 || ((reset_cnt >> (x-1)) & 1) == 0) {
131  /* terminate this hash */
132  if ((err = sha256_done(&prng->u.fortuna.pool[x], tmp)) != CRYPT_OK) {
133  sha256_done(&md, tmp);
134  return err;
135  }
136  /* add it to the string */
137  if ((err = sha256_process(&md, tmp, 32)) != CRYPT_OK) {
138  sha256_done(&md, tmp);
139  return err;
140  }
141  /* reset this pool */
142  if ((err = sha256_init(&prng->u.fortuna.pool[x])) != CRYPT_OK) {
143  sha256_done(&md, tmp);
144  return err;
145  }
146  } else {
147  break;
148  }
149  }
150 
151  /* finish key */
152  if ((err = sha256_done(&md, prng->u.fortuna.K)) != CRYPT_OK) {
153  return err;
154  }
155  if ((err = rijndael_setup(prng->u.fortuna.K, 32, 0, &prng->u.fortuna.skey)) != CRYPT_OK) {
156  return err;
157  }
158  _fortuna_update_iv(prng);
159 
160  /* reset/update internals */
161  prng->u.fortuna.pool0_len = 0;
162 #ifdef LTC_FORTUNA_RESEED_RATELIMIT_TIMED
163  prng->u.fortuna.wd = now;
164 #else
165  prng->u.fortuna.wd = 0;
166 #endif
167  prng->u.fortuna.reset_cnt = reset_cnt;
168 
169 
170 #ifdef LTC_CLEAN_STACK
171  zeromem(&md, sizeof(md));
172  zeromem(tmp, sizeof(tmp));
173 #endif
174 
175  return CRYPT_OK;
176 }

179  "Update Seed File"-compliant update of K
180 
181  @param in The PRNG state
182  @param inlen Size of the state
183  @param prng The PRNG to import
184  @return CRYPT_OK if successful

186 int fortuna_update_seed(const unsigned char *in, unsigned long inlen, prng_state *prng)
187 {
188  int err;
189  unsigned char tmp[MAXBLOCKSIZE];
190  hash_state md;
191 
192  LTC_MUTEX_LOCK(&prng->lock);
193  /* new K = LTC_SHA256(K || in) */
194  sha256_init(&md);
195  if ((err = sha256_process(&md, prng->u.fortuna.K, 32)) != CRYPT_OK) {
196  sha256_done(&md, tmp);
197  goto LBL_UNLOCK;
198  }
199  if ((err = sha256_process(&md, in, inlen)) != CRYPT_OK) {
200  sha256_done(&md, tmp);
201  goto LBL_UNLOCK;
202  }
203  /* finish key */
204  if ((err = sha256_done(&md, prng->u.fortuna.K)) != CRYPT_OK) {
205  goto LBL_UNLOCK;
206  }
207  _fortuna_update_iv(prng);
208 
209 LBL_UNLOCK:
210  LTC_MUTEX_UNLOCK(&prng->lock);

212  zeromem(&md, sizeof(md));

218 /**
219  Start the PRNG
220  @param prng [out] The PRNG state to initialize
221  @return CRYPT_OK if successful
222 */
223 int fortuna_start(prng_state *prng)
224 {
225  int err, x, y;
226  unsigned char tmp[MAXBLOCKSIZE];
227 
228  LTC_ARGCHK(prng != NULL);
229  prng->ready = 0;
230 
231  /* initialize the pools */
232  for (x = 0; x < LTC_FORTUNA_POOLS; x++) {
233  if ((err = sha256_init(&prng->u.fortuna.pool[x])) != CRYPT_OK) {
234  for (y = 0; y < x; y++) {
235  sha256_done(&prng->u.fortuna.pool[y], tmp);
236  }
237  return err;
238  }
239  }
240  prng->u.fortuna.pool_idx = prng->u.fortuna.pool0_len = prng->u.fortuna.wd = 0;
241  prng->u.fortuna.reset_cnt = 0;
242 
243  /* reset bufs */
244  zeromem(prng->u.fortuna.K, 32);
245  if ((err = rijndael_setup(prng->u.fortuna.K, 32, 0, &prng->u.fortuna.skey)) != CRYPT_OK) {
246  for (x = 0; x < LTC_FORTUNA_POOLS; x++) {
247  sha256_done(&prng->u.fortuna.pool[x], tmp);
248  }
249  return err;
250  }
251  zeromem(prng->u.fortuna.IV, 16);
252 
253  LTC_MUTEX_INIT(&prng->lock)
254 
255  return CRYPT_OK;
256 }

258 static int _fortuna_add(unsigned long source, unsigned long pool, const unsigned char *in, unsigned long inlen, prng_state *prng)
259 {
260  unsigned char tmp[2];
261  int err;
262 
263  /* ensure inlen <= 32 */
264  if (inlen > 32) {
265  inlen = 32;
266  }
267 
268  /* add s || length(in) || in to pool[pool_idx] */
269  tmp[0] = (unsigned char)source;
270  tmp[1] = (unsigned char)inlen;
271 
272  if ((err = sha256_process(&prng->u.fortuna.pool[pool], tmp, 2)) != CRYPT_OK) {
273  return err;
274  }
275  if ((err = sha256_process(&prng->u.fortuna.pool[pool], in, inlen)) != CRYPT_OK) {
276  return err;
277  }
278  if (pool == 0) {
279  prng->u.fortuna.pool0_len += inlen;
280  }
281  return CRYPT_OK; /* success */
282 }

350

366 /**
367  Read from the PRNG
368  @param out Destination
369  @param outlen Length of output
370  @param prng The active PRNG to read from
371  @return Number of octets read
372 */
373 unsigned long fortuna_read(unsigned char *out, unsigned long outlen, prng_state *prng)
374 {
375  unsigned char tmp[16];
376  unsigned long tlen = 0;
377 
378  if (outlen == 0 || prng == NULL || out == NULL) return 0;
379 
380  LTC_MUTEX_LOCK(&prng->lock);
381 
382  if (!prng->ready) {
383  goto LBL_UNLOCK;
384  }
385 
386  /* do we have to reseed? */
387  if (prng->u.fortuna.pool0_len >= 64) {
388  if (_fortuna_reseed(prng) != CRYPT_OK) {
389  goto LBL_UNLOCK;
390  }
391  }
392 
393  /* ensure that one reseed happened before allowing to read */
394  if (prng->u.fortuna.reset_cnt == 0) {
395  goto LBL_UNLOCK;
396  }
397 
398  /* now generate the blocks required */
399  tlen = outlen;
400 
401  /* handle whole blocks without the extra XMEMCPY */
402  while (outlen >= 16) {
403  /* encrypt the IV and store it */
404  rijndael_ecb_encrypt(prng->u.fortuna.IV, out, &prng->u.fortuna.skey);
405  out += 16;
406  outlen -= 16;
407  _fortuna_update_iv(prng);
408  }
409 
410  /* left over bytes? */
411  if (outlen > 0) {
412  rijndael_ecb_encrypt(prng->u.fortuna.IV, tmp, &prng->u.fortuna.skey);
413  XMEMCPY(out, tmp, outlen);
414  _fortuna_update_iv(prng);
415  }
416 
417  /* generate new key */
418  rijndael_ecb_encrypt(prng->u.fortuna.IV, prng->u.fortuna.K , &prng->u.fortuna.skey);
419  _fortuna_update_iv(prng);
420 
421  rijndael_ecb_encrypt(prng->u.fortuna.IV, prng->u.fortuna.K+16, &prng->u.fortuna.skey);
422  _fortuna_update_iv(prng);
423 
424  if (rijndael_setup(prng->u.fortuna.K, 32, 0, &prng->u.fortuna.skey) != CRYPT_OK) {
425  tlen = 0;
426  }
427 
428 LBL_UNLOCK:
429 #ifdef LTC_CLEAN_STACK
430  zeromem(tmp, sizeof(tmp));
431 #endif
432  LTC_MUTEX_UNLOCK(&prng->lock);
433  return tlen;
434 }
