186 static int blake2b_init_param(hash_state *md, const unsigned char *P)
187 {
188  unsigned long i;
189 
190  blake2b_init0(md);
191 
192  /* IV XOR ParamBlock */
193  for (i = 0; i < 8; ++i) {
194  ulong64 tmp;
195  LOAD64L(tmp, P + i * 8);
196  md->blake2b.h[i] ^= tmp;
197  }
198 
199  md->blake2b.outlen = P[O_DIGEST_LENGTH];
200  return CRYPT_OK;

203 /**
204  Initialize the hash/MAC state
205 
206  Use this function to init for arbitrary sizes.
207 
208  Give a key and keylen to init for MAC mode.
209 
210  @param md The hash state you wish to initialize
211  @param outlen The desired output-length
212  @param key The key of the MAC
213  @param keylen The length of the key
214  @return CRYPT_OK if successful
215 */
216 int blake2b_init(hash_state *md, unsigned long outlen, const unsigned char *key, unsigned long keylen)
217 {
218  unsigned char P[BLAKE2B_PARAM_SIZE];
219  int err;
220 
221  LTC_ARGCHK(md != NULL);
222 
223  if ((!outlen) || (outlen > BLAKE2B_OUTBYTES)) {
224  return CRYPT_INVALID_ARG;
225  }
226  if ((key && !keylen) || (keylen && !key) || (keylen > BLAKE2B_KEYBYTES)) {
227  return CRYPT_INVALID_ARG;
228  }
229 
230  XMEMSET(P, 0, sizeof(P));
231 
232  P[O_DIGEST_LENGTH] = (unsigned char)outlen;
233  P[O_KEY_LENGTH] = (unsigned char)keylen;
234  P[O_FANOUT] = 1;
235  P[O_DEPTH] = 1;
236 
237  err = blake2b_init_param(md, P);
238  if (err != CRYPT_OK) return err;
239 
240  if (key) {
241  unsigned char block[BLAKE2B_BLOCKBYTES];
242 
243  XMEMSET(block, 0, BLAKE2B_BLOCKBYTES);
244  XMEMCPY(block, key, keylen);
245  blake2b_process(md, block, BLAKE2B_BLOCKBYTES);
246 
247 #ifdef LTC_CLEAN_STACK
248  zeromem(block, sizeof(block));
249 #endif
250  }
251 
252  return CRYPT_OK;
253 }

308 static int _blake2b_compress(hash_state *md, const unsigned char *buf)
309 #else
310 static int blake2b_compress(hash_state *md, const unsigned char *buf)
311 #endif
312 {
313  ulong64 m[16];
314  ulong64 v[16];
315  unsigned long i;
316 
317  for (i = 0; i < 16; ++i) {
318  LOAD64L(m[i], buf + i * sizeof(m[i]));
319  }
320 
321  for (i = 0; i < 8; ++i) {
322  v[i] = md->blake2b.h[i];
323  }
324 
325  v[8] = blake2b_IV[0];
326  v[9] = blake2b_IV[1];
327  v[10] = blake2b_IV[2];
328  v[11] = blake2b_IV[3];
329  v[12] = blake2b_IV[4] ^ md->blake2b.t[0];
330  v[13] = blake2b_IV[5] ^ md->blake2b.t[1];
331  v[14] = blake2b_IV[6] ^ md->blake2b.f[0];
332  v[15] = blake2b_IV[7] ^ md->blake2b.f[1];
333 
334  ROUND(0);
335  ROUND(1);
336  ROUND(2);
337  ROUND(3);
338  ROUND(4);
339  ROUND(5);
340  ROUND(6);
341  ROUND(7);
342  ROUND(8);
343  ROUND(9);
344  ROUND(10);
345  ROUND(11);
346 
347  for (i = 0; i < 8; ++i) {
348  md->blake2b.h[i] = md->blake2b.h[i] ^ v[i] ^ v[i + 8];
349  }
350  return CRYPT_OK;
351 }

366 /**
367  Process a block of memory through the hash
368  @param md The hash state
369  @param in The data to hash
370  @param inlen The length of the data (octets)
371  @return CRYPT_OK if successful
372 */
373 int blake2b_process(hash_state *md, const unsigned char *in, unsigned long inlen)
374 {
375  LTC_ARGCHK(md != NULL);
376  LTC_ARGCHK(in != NULL);
377 
378  if (md->blake2b.curlen > sizeof(md->blake2b.buf)) {
379  return CRYPT_INVALID_ARG;
380  }
381 
382  if (inlen > 0) {
383  unsigned long left = md->blake2b.curlen;
384  unsigned long fill = BLAKE2B_BLOCKBYTES - left;
385  if (inlen > fill) {
386  md->blake2b.curlen = 0;
387  XMEMCPY(md->blake2b.buf + (left % sizeof(md->blake2b.buf)), in, fill); /* Fill buffer */
388  blake2b_increment_counter(md, BLAKE2B_BLOCKBYTES);
389  blake2b_compress(md, md->blake2b.buf); /* Compress */
390  in += fill;
391  inlen -= fill;
392  while (inlen > BLAKE2B_BLOCKBYTES) {
393  blake2b_increment_counter(md, BLAKE2B_BLOCKBYTES);
394  blake2b_compress(md, in);
395  in += BLAKE2B_BLOCKBYTES;
396  inlen -= BLAKE2B_BLOCKBYTES;
397  }
398  }
399  XMEMCPY(md->blake2b.buf + md->blake2b.curlen, in, inlen);
400  md->blake2b.curlen += inlen;
401  }
402  return CRYPT_OK;
403 }

405 /**
406  Terminate the hash to get the digest
407  @param md The hash state
408  @param out [out] The destination of the hash (size depending on the length used on init)
409  @return CRYPT_OK if successful
410 */
411 int blake2b_done(hash_state *md, unsigned char *out)
412 {
413  unsigned char buffer[BLAKE2B_OUTBYTES] = { 0 };
414  unsigned long i;
415 
416  LTC_ARGCHK(md != NULL);
417  LTC_ARGCHK(out != NULL);
418 
419  /* if(md->blakebs.outlen != outlen) return CRYPT_INVALID_ARG; */
420 
421  if (blake2b_is_lastblock(md)) {
422  return CRYPT_ERROR;
423  }
424 
425  blake2b_increment_counter(md, md->blake2b.curlen);
426  blake2b_set_lastblock(md);
427  XMEMSET(md->blake2b.buf + md->blake2b.curlen, 0, BLAKE2B_BLOCKBYTES - md->blake2b.curlen); /* Padding */
428  blake2b_compress(md, md->blake2b.buf);
429 
430  for (i = 0; i < 8; ++i) { /* Output full hash to temp buffer */
431  STORE64L(md->blake2b.h[i], buffer + i * 8);
432  }
433 
434  XMEMCPY(out, buffer, md->blake2b.outlen);
435  zeromem(md, sizeof(hash_state));
436 #ifdef LTC_CLEAN_STACK
437  zeromem(buffer, sizeof(buffer));
438 #endif
439  return CRYPT_OK;
440 }
