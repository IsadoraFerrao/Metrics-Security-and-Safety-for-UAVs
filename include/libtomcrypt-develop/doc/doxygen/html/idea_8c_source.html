42 typedef unsigned short int ushort16;
43 
44 #define _LOW16(x) ((x)&0xffff) /* compiler should be able to optimize this away if x is 16 bits */
45 #define _HIGH16(x) ((x)>>16)
46 #define _MUL(a,b) { \
47  ulong32 p = (ulong32)_LOW16(a) * b; \
48  if (p) { \
49  p = _LOW16(p) - _HIGH16(p); \
50  a = (ushort16)p - (ushort16)_HIGH16(p); \
51  } \
52  else \
53  a = 1 - a - b; \
54  }
55 #define _STORE16(x,y) { (y)[0] = (unsigned char)(((x)>>8)&255); (y)[1] = (unsigned char)((x)&255); }
56 #define _LOAD16(x,y) { x = ((ushort16)((y)[0] & 255)<<8) | ((ushort16)((y)[1] & 255)); }

58 static ushort16 _mul_inv(ushort16 x)
59 {
60  ushort16 y = x;
61  unsigned i;
62 
63  for (i = 0; i < 15; i++) {
64  _MUL(y, _LOW16(y));
65  _MUL(y, x);
66  }
67  return _LOW16(y);
68 }

75 static int _setup_key(const unsigned char *key, symmetric_key *skey)
76 {
77  int i, j;
78  ushort16 *e_key = skey->idea.ek;
79  ushort16 *d_key = skey->idea.dk;
80 
81  /* prepare enc key */
82  for (i = 0; i < 8; i++) {
83  _LOAD16(e_key[i], key + 2 * i);
84  }
85  for (; i < LTC_IDEA_KEYLEN; i++) {
86  j = (i - i % 8) - 8;
87  e_key[i] = _LOW16((e_key[j+(i+1)%8] << 9) | (e_key[j+(i+2)%8] >> 7));
88  }
89 
90  /* prepare dec key */
91  for (i = 0; i < LTC_IDEA_ROUNDS; i++) {
92  d_key[i*6+0] = _mul_inv(e_key[(LTC_IDEA_ROUNDS-i)*6+0]);
93  d_key[i*6+1] = _add_inv(e_key[(LTC_IDEA_ROUNDS-i)*6+1+(i>0 ? 1 : 0)]);
94  d_key[i*6+2] = _add_inv(e_key[(LTC_IDEA_ROUNDS-i)*6+2-(i>0 ? 1 : 0)]);
95  d_key[i*6+3] = _mul_inv(e_key[(LTC_IDEA_ROUNDS-i)*6+3]);
96  d_key[i*6+4] = e_key[(LTC_IDEA_ROUNDS-1-i)*6+4];
97  d_key[i*6+5] = e_key[(LTC_IDEA_ROUNDS-1-i)*6+5];
98  }
99  d_key[i*6+0] = _mul_inv(e_key[(LTC_IDEA_ROUNDS-i)*6+0]);
100  d_key[i*6+1] = _add_inv(e_key[(LTC_IDEA_ROUNDS-i)*6+1]);
101  d_key[i*6+2] = _add_inv(e_key[(LTC_IDEA_ROUNDS-i)*6+2]);
102  d_key[i*6+3] = _mul_inv(e_key[(LTC_IDEA_ROUNDS-i)*6+3]);
103 
104  return CRYPT_OK;
105 }

107 static int _process_block(const unsigned char *in, unsigned char *out, const ushort16 *m_key)
108 {
109  int i;
110  ushort16 x0, x1, x2, x3, t0, t1;
111 
112  _LOAD16(x0, in + 0);
113  _LOAD16(x1, in + 2);
114  _LOAD16(x2, in + 4);
115  _LOAD16(x3, in + 6);
116 
117  for (i = 0; i < LTC_IDEA_ROUNDS; i++) {
118  _MUL(x0, m_key[i*6+0]);
119  x1 += m_key[i*6+1];
120  x2 += m_key[i*6+2];
121  _MUL(x3, m_key[i*6+3]);
122  t0 = x0^x2;
123  _MUL(t0, m_key[i*6+4]);
124  t1 = t0 + (x1^x3);
125  _MUL(t1, m_key[i*6+5]);
126  t0 += t1;
127  x0 ^= t1;
128  x3 ^= t0;
129  t0 ^= x1;
130  x1 = x2^t1;
131  x2 = t0;
132  }
133 
134  _MUL(x0, m_key[LTC_IDEA_ROUNDS*6+0]);
135  x2 += m_key[LTC_IDEA_ROUNDS*6+1];
136  x1 += m_key[LTC_IDEA_ROUNDS*6+2];
137  _MUL(x3, m_key[LTC_IDEA_ROUNDS*6+3]);
138 
139  _STORE16(x0, out + 0);
140  _STORE16(x2, out + 2);
141  _STORE16(x1, out + 4);
142  _STORE16(x3, out + 6);
143 
144  return CRYPT_OK;
145 }
