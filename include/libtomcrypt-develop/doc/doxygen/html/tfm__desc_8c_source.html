28  @return The equivalent LTC error code or CRYPT_ERROR if none found

31 {
32  int x;
33 
34  for (x = 0; x < (int)(sizeof(tfm_to_ltc_codes)/sizeof(tfm_to_ltc_codes[0])); x++) {
35  if (err == tfm_to_ltc_codes[x].tfm_code) {

429 static int tfm_ecc_projective_dbl_point(const ecc_point *P, ecc_point *R, void *ma, void *modulus, void *Mp)
430 {
431  fp_int t1, t2;
432  fp_digit mp;
433  int err, inf;
434 
435  LTC_ARGCHK(P != NULL);
436  LTC_ARGCHK(R != NULL);
437  LTC_ARGCHK(modulus != NULL);
438  LTC_ARGCHK(Mp != NULL);
439 
440  mp = *((fp_digit*)Mp);
441 
442  fp_init(&t1);
443  fp_init(&t2);
444 
445  if (P != R) {
446  fp_copy(P->x, R->x);
447  fp_copy(P->y, R->y);
448  fp_copy(P->z, R->z);
449  }
450 
451  if ((err = ltc_ecc_is_point_at_infinity(P, modulus, &inf)) != CRYPT_OK) return err;
452  if (inf) {
453  /* if P is point at infinity >> Result = point at infinity */
454  ltc_mp.set_int(R->x, 1);
455  ltc_mp.set_int(R->y, 1);
456  ltc_mp.set_int(R->z, 0);
457  return CRYPT_OK;
458  }
459 
460  /* t1 = Z * Z */
461  fp_sqr(R->z, &t1);
462  fp_montgomery_reduce(&t1, modulus, mp);
463  /* Z = Y * Z */
464  fp_mul(R->z, R->y, R->z);
465  fp_montgomery_reduce(R->z, modulus, mp);
466  /* Z = 2Z */
467  fp_add(R->z, R->z, R->z);
468  if (fp_cmp(R->z, modulus) != FP_LT) {
469  fp_sub(R->z, modulus, R->z);
470  }
471 
472  if (ma == NULL) { /* special case for curves with a == -3 (10% faster than general case) */
473  /* T2 = X - T1 */
474  fp_sub(R->x, &t1, &t2);
475  if (fp_cmp_d(&t2, 0) == LTC_MP_LT) {
476  fp_add(&t2, modulus, &t2);
477  }
478  /* T1 = X + T1 */
479  fp_add(&t1, R->x, &t1);
480  if (fp_cmp(&t1, modulus) != FP_LT) {
481  fp_sub(&t1, modulus, &t1);
482  }
483  /* T2 = T1 * T2 */
484  fp_mul(&t1, &t2, &t2);
485  fp_montgomery_reduce(&t2, modulus, mp);
486  /* T1 = 2T2 */
487  fp_add(&t2, &t2, &t1);
488  if (fp_cmp(&t1, modulus) != FP_LT) {
489  fp_sub(&t1, modulus, &t1);
490  }
491  /* T1 = T1 + T2 */
492  fp_add(&t1, &t2, &t1);
493  if (fp_cmp(&t1, modulus) != FP_LT) {
494  fp_sub(&t1, modulus, &t1);
495  }
496  }
497  else {
498  /* T2 = T1 * T1 */
499  fp_sqr(&t1, &t2);
500  fp_montgomery_reduce(&t2, modulus, mp);
501  /* T1 = T2 * a */
502  fp_mul(&t2, ma, &t1);
503  fp_montgomery_reduce(&t1, modulus, mp);
504  /* T2 = X * X */
505  fp_sqr(R->x, &t2);
506  fp_montgomery_reduce(&t2, modulus, mp);
507  /* T1 = T1 + T2 */
508  fp_add(&t1, &t2, &t1);
509  if (fp_cmp(&t1, modulus) != FP_LT) {
510  fp_sub(&t1, modulus, &t1);
511  }
512  /* T1 = T1 + T2 */
513  fp_add(&t1, &t2, &t1);
514  if (fp_cmp(&t1, modulus) != FP_LT) {
515  fp_sub(&t1, modulus, &t1);
516  }
517  /* T1 = T1 + T2 */
518  fp_add(&t1, &t2, &t1);
519  if (fp_cmp(&t1, modulus) != FP_LT) {
520  fp_sub(&t1, modulus, &t1);
521  }
522  }
523 
524  /* Y = 2Y */
525  fp_add(R->y, R->y, R->y);
526  if (fp_cmp(R->y, modulus) != FP_LT) {
527  fp_sub(R->y, modulus, R->y);
528  }
529  /* Y = Y * Y */
530  fp_sqr(R->y, R->y);
531  fp_montgomery_reduce(R->y, modulus, mp);
532  /* T2 = Y * Y */
533  fp_sqr(R->y, &t2);
534  fp_montgomery_reduce(&t2, modulus, mp);
535  /* T2 = T2/2 */
536  if (fp_isodd(&t2)) {
537  fp_add(&t2, modulus, &t2);
538  }
539  fp_div_2(&t2, &t2);
540  /* Y = Y * X */
541  fp_mul(R->y, R->x, R->y);
542  fp_montgomery_reduce(R->y, modulus, mp);
543 
544  /* X = T1 * T1 */
545  fp_sqr(&t1, R->x);
546  fp_montgomery_reduce(R->x, modulus, mp);
547  /* X = X - Y */
548  fp_sub(R->x, R->y, R->x);
549  if (fp_cmp_d(R->x, 0) == FP_LT) {
550  fp_add(R->x, modulus, R->x);
551  }
552  /* X = X - Y */
553  fp_sub(R->x, R->y, R->x);
554  if (fp_cmp_d(R->x, 0) == FP_LT) {
555  fp_add(R->x, modulus, R->x);
556  }
557 
558  /* Y = Y - X */
559  fp_sub(R->y, R->x, R->y);
560  if (fp_cmp_d(R->y, 0) == FP_LT) {
561  fp_add(R->y, modulus, R->y);
562  }
563  /* Y = Y * T1 */
564  fp_mul(R->y, &t1, R->y);
565  fp_montgomery_reduce(R->y, modulus, mp);
566  /* Y = Y - T2 */
567  fp_sub(R->y, &t2, R->y);
568  if (fp_cmp_d(R->y, 0) == FP_LT) {
569  fp_add(R->y, modulus, R->y);
570  }
571 
572  return CRYPT_OK;
573 }

575 /**
576  Add two ECC points
577  @param P The point to add
578  @param Q The point to add
579  @param R [out] The destination of the double
580  @param modulus The modulus of the field the ECC curve is in
581  @param Mp The "b" value from montgomery_setup()
582  @return CRYPT_OK on success
583 */
584 static int tfm_ecc_projective_add_point(const ecc_point *P, const ecc_point *Q, ecc_point *R, void *ma, void *modulus, void *Mp)
585 {
586  fp_int t1, t2, x, y, z;
587  fp_digit mp;
588  int err, inf;
589 
590  LTC_ARGCHK(P != NULL);
591  LTC_ARGCHK(Q != NULL);
592  LTC_ARGCHK(R != NULL);
593  LTC_ARGCHK(modulus != NULL);
594  LTC_ARGCHK(Mp != NULL);
595 
596  mp = *((fp_digit*)Mp);
597 
598  fp_init(&t1);
599  fp_init(&t2);
600  fp_init(&x);
601  fp_init(&y);
602  fp_init(&z);
603 
604  if ((err = ltc_ecc_is_point_at_infinity(P, modulus, &inf)) != CRYPT_OK) return err;
605  if (inf) {
606  /* P is point at infinity >> Result = Q */
607  ltc_mp.copy(Q->x, R->x);
608  ltc_mp.copy(Q->y, R->y);
609  ltc_mp.copy(Q->z, R->z);
610  return CRYPT_OK;
611  }
612 
613  if ((err = ltc_ecc_is_point_at_infinity(Q, modulus, &inf)) != CRYPT_OK) return err;
614  if (inf) {
615  /* Q is point at infinity >> Result = P */
616  ltc_mp.copy(P->x, R->x);
617  ltc_mp.copy(P->y, R->y);
618  ltc_mp.copy(P->z, R->z);
619  return CRYPT_OK;
620  }
621 
622  /* should we dbl instead? */
623  fp_sub(modulus, Q->y, &t1);
624  if ( (fp_cmp(P->x, Q->x) == FP_EQ) &&
625  (Q->z != NULL && fp_cmp(P->z, Q->z) == FP_EQ) &&
626  (fp_cmp(P->y, Q->y) == FP_EQ || fp_cmp(P->y, &t1) == FP_EQ)) {
627  return tfm_ecc_projective_dbl_point(P, R, ma, modulus, Mp);
628  }
629 
630  fp_copy(P->x, &x);
631  fp_copy(P->y, &y);
632  fp_copy(P->z, &z);
633 
634  /* if Z is one then these are no-operations */
635  if (Q->z != NULL) {
636  /* T1 = Z' * Z' */
637  fp_sqr(Q->z, &t1);
638  fp_montgomery_reduce(&t1, modulus, mp);
639  /* X = X * T1 */
640  fp_mul(&t1, &x, &x);
641  fp_montgomery_reduce(&x, modulus, mp);
642  /* T1 = Z' * T1 */
643  fp_mul(Q->z, &t1, &t1);
644  fp_montgomery_reduce(&t1, modulus, mp);
645  /* Y = Y * T1 */
646  fp_mul(&t1, &y, &y);
647  fp_montgomery_reduce(&y, modulus, mp);
648  }
649 
650  /* T1 = Z*Z */
651  fp_sqr(&z, &t1);
652  fp_montgomery_reduce(&t1, modulus, mp);
653  /* T2 = X' * T1 */
654  fp_mul(Q->x, &t1, &t2);
655  fp_montgomery_reduce(&t2, modulus, mp);
656  /* T1 = Z * T1 */
657  fp_mul(&z, &t1, &t1);
658  fp_montgomery_reduce(&t1, modulus, mp);
659  /* T1 = Y' * T1 */
