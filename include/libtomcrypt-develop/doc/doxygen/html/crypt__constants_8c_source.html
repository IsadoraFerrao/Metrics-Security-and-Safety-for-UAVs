232 /* crypt_get_constant()
233  * valueout will be the value of the named constant
234  * return -1 if named item not found
235  */
236 int crypt_get_constant(const char* namein, int *valueout) {
237  int i;
238  int _crypt_constants_len = sizeof(_crypt_constants) / sizeof(_crypt_constants[0]);
239  for (i=0; i<_crypt_constants_len; i++) {
240  if (XSTRCMP(_crypt_constants[i].name, namein) == 0) {
241  *valueout = _crypt_constants[i].value;
242  return 0;
243  }
244  }
245  return 1;
246 }

248 /* crypt_list_all_constants()
249  * if names_list is NULL, names_list_size will be the minimum
250  * number of bytes needed to receive the complete names_list
251  * if names_list is NOT NULL, names_list must be the addr of
252  * sufficient memory allocated into which the names_list
253  * is to be written. Also, the value in names_list_size
254  * sets the upper bound of the number of characters to be
255  * written.
256  * a -1 return value signifies insufficient space made available
257  */
258 int crypt_list_all_constants(char *names_list, unsigned int *names_list_size) {
259  int i;
260  unsigned int total_len = 0;
261  char *ptr;
262  int number_len;
263  int count = sizeof(_crypt_constants) / sizeof(_crypt_constants[0]);
264 
265  /* calculate amount of memory required for the list */
266  for (i=0; i<count; i++) {
267  number_len = snprintf(NULL, 0, "%s,%d\n", _crypt_constants[i].name, _crypt_constants[i].value);
268  if (number_len < 0) {
269  return -1;
270  }
271  total_len += number_len;
272  }
273 
274  if (names_list == NULL) {
275  *names_list_size = total_len;
276  } else {
277  if (total_len > *names_list_size) {
278  return -1;
279  }
280  /* build the names list */
281  ptr = names_list;
282  for (i=0; i<count; i++) {
283  number_len = snprintf(ptr, total_len, "%s,%d\n", _crypt_constants[i].name, _crypt_constants[i].value);
284  if (number_len < 0) return -1;
285  if ((unsigned int)number_len > total_len) return -1;
286  total_len -= number_len;
287  ptr += number_len;
288  }
289  /* to remove the trailing new-line */
290  ptr -= 1;
291  *ptr = 0;
292  }
293  return 0;
294 }
