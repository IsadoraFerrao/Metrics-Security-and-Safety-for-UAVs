452 static int _setup_key(const unsigned char *key, int keylen, int rounds, ulong32 *k)
453 {
454  int i;
455  ulong32 t;
456  ulong32 k0[8] = { 0 }; /* zero-initialize */
457  ulong32 a, b, c, d, e;
458 
459  for (i = 0; i < 8 && i < keylen/4; ++i) {
460  LOAD32L(k0[i], key + i * 4);
461  }
462  if (keylen < 32) {
463  k0[keylen/4] |= (ulong32)1 << ((keylen%4)*8);
464  }
465 
466  t = k0[7];
467  for (i = 0; i < 8; ++i) {
468  k[i] = k0[i] = t = ROLc(k0[i] ^ k0[(i+3)%8] ^ k0[(i+5)%8] ^ t ^ 0x9e3779b9 ^ i, 11);
469  }
470  for (i = 8; i < 4*(rounds+1); ++i) {
471  k[i] = t = ROLc(k[i-8] ^ k[i-5] ^ k[i-3] ^ t ^ 0x9e3779b9 ^ i, 11);
472  }
473  k -= 20;
474 
475  for (i = 0; i < rounds/8; i++) {
476  _afterS2(_LK); _afterS2(_S3); _afterS3(_SK);
477  _afterS1(_LK); _afterS1(_S2); _afterS2(_SK);
478  _afterS0(_LK); _afterS0(_S1); _afterS1(_SK);
479  _beforeS0(_LK); _beforeS0(_S0); _afterS0(_SK);
480  k += 8*4;
481  _afterS6(_LK); _afterS6(_S7); _afterS7(_SK);
482  _afterS5(_LK); _afterS5(_S6); _afterS6(_SK);
483  _afterS4(_LK); _afterS4(_S5); _afterS5(_SK);
484  _afterS3(_LK); _afterS3(_S4); _afterS4(_SK);
485  }
486  _afterS2(_LK); _afterS2(_S3); _afterS3(_SK);
487 
488  return CRYPT_OK;
489 }

491 static int _enc_block(const unsigned char *in, unsigned char *out, const ulong32 *k)
492 {
493  ulong32 a, b, c, d, e;
494  unsigned int i = 1;
495 
496  LOAD32L(a, in + 0);
497  LOAD32L(b, in + 4);
498  LOAD32L(c, in + 8);
499  LOAD32L(d, in + 12);
500 
501  do {
502  _beforeS0(_KX); _beforeS0(_S0); _afterS0(_LT);
503  _afterS0(_KX); _afterS0(_S1); _afterS1(_LT);
504  _afterS1(_KX); _afterS1(_S2); _afterS2(_LT);
505  _afterS2(_KX); _afterS2(_S3); _afterS3(_LT);
506  _afterS3(_KX); _afterS3(_S4); _afterS4(_LT);
507  _afterS4(_KX); _afterS4(_S5); _afterS5(_LT);
508  _afterS5(_KX); _afterS5(_S6); _afterS6(_LT);
509  _afterS6(_KX); _afterS6(_S7);
510 
511  if (i == 4) break;
512 
513  ++i;
514  c = b;
515  b = e;
516  e = d;
517  d = a;
518  a = e;
519  k += 32;
520  _beforeS0(_LT);
521  } while (1);
522 
523  _afterS7(_KX);
524 
525  STORE32L(d, out + 0);
526  STORE32L(e, out + 4);
527  STORE32L(b, out + 8);
528  STORE32L(a, out + 12);
529 
530  return CRYPT_OK;
531 }

534 {
