109  int strengthened,

111 #endif
112 { unsigned int i, j, k;
113  unsigned char ka[LTC_SAFER_BLOCK_LEN + 1];
114  unsigned char kb[LTC_SAFER_BLOCK_LEN + 1];
115 
116  if (LTC_SAFER_MAX_NOF_ROUNDS < nof_rounds) {
117  nof_rounds = LTC_SAFER_MAX_NOF_ROUNDS;
118  }
119  *key++ = (unsigned char)nof_rounds;
120  ka[LTC_SAFER_BLOCK_LEN] = (unsigned char)0;
121  kb[LTC_SAFER_BLOCK_LEN] = (unsigned char)0;
122  k = 0;
123  for (j = 0; j < LTC_SAFER_BLOCK_LEN; j++) {
124  ka[j] = ROL8(userkey_1[j], 5);
125  ka[LTC_SAFER_BLOCK_LEN] ^= ka[j];
126  kb[j] = *key++ = userkey_2[j];
127  kb[LTC_SAFER_BLOCK_LEN] ^= kb[j];
128  }
129  for (i = 1; i <= nof_rounds; i++) {
130  for (j = 0; j < LTC_SAFER_BLOCK_LEN + 1; j++) {
131  ka[j] = ROL8(ka[j], 6);
132  kb[j] = ROL8(kb[j], 6);
133  }
134  if (strengthened) {
135  k = 2 * i - 1;
136  while (k >= (LTC_SAFER_BLOCK_LEN + 1)) { k -= LTC_SAFER_BLOCK_LEN + 1; }
137  }
138  for (j = 0; j < LTC_SAFER_BLOCK_LEN; j++) {
139  if (strengthened) {
140  *key++ = (ka[k]
141  + safer_ebox[(int)safer_ebox[(int)((18 * i + j + 1)&0xFF)]]) & 0xFF;
142  if (++k == (LTC_SAFER_BLOCK_LEN + 1)) { k = 0; }
143  } else {
144  *key++ = (ka[j] + safer_ebox[(int)safer_ebox[(int)((18 * i + j + 1)&0xFF)]]) & 0xFF;
145  }
146  }
147  if (strengthened) {
148  k = 2 * i;
149  while (k >= (LTC_SAFER_BLOCK_LEN + 1)) { k -= LTC_SAFER_BLOCK_LEN + 1; }
150  }
151  for (j = 0; j < LTC_SAFER_BLOCK_LEN; j++) {
152  if (strengthened) {
153  *key++ = (kb[k]
154  + safer_ebox[(int)safer_ebox[(int)((18 * i + j + 10)&0xFF)]]) & 0xFF;
155  if (++k == (LTC_SAFER_BLOCK_LEN + 1)) { k = 0; }
156  } else {

158  }

160  }
