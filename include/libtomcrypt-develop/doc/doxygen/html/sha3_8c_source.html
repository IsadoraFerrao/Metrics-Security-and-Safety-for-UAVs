163 static void keccakf(ulong64 s[25])
164 {
165  int i, j, round;
166  ulong64 t, bc[5];
167 
168  for(round = 0; round < SHA3_KECCAK_ROUNDS; round++) {
169  /* Theta */
170  for(i = 0; i < 5; i++) {
171  bc[i] = s[i] ^ s[i + 5] ^ s[i + 10] ^ s[i + 15] ^ s[i + 20];
172  }
173  for(i = 0; i < 5; i++) {
174  t = bc[(i + 4) % 5] ^ ROL64(bc[(i + 1) % 5], 1);
175  for(j = 0; j < 25; j += 5) {
176  s[j + i] ^= t;
177  }
178  }
179  /* Rho Pi */
180  t = s[1];
181  for(i = 0; i < 24; i++) {
182  j = keccakf_piln[i];
183  bc[0] = s[j];
184  s[j] = ROL64(t, keccakf_rotc[i]);
185  t = bc[0];
186  }
187  /* Chi */
188  for(j = 0; j < 25; j += 5) {
189  for(i = 0; i < 5; i++) {
190  bc[i] = s[j + i];
191  }
192  for(i = 0; i < 5; i++) {
193  s[j + i] ^= (~bc[(i + 1) % 5]) & bc[(i + 2) % 5];
194  }
195  }
196  /* Iota */
197  s[0] ^= keccakf_rndc[round];
198  }
199 }

201 static LTC_INLINE int _done(hash_state *md, unsigned char *hash, ulong64 pad)
202 {
203  unsigned i;
204 
205  LTC_ARGCHK(md != NULL);
206  LTC_ARGCHK(hash != NULL);
207 
208  md->sha3.s[md->sha3.word_index] ^= (md->sha3.saved ^ (pad << (md->sha3.byte_index * 8)));
209  md->sha3.s[SHA3_KECCAK_SPONGE_WORDS - md->sha3.capacity_words - 1] ^= CONST64(0x8000000000000000);
210  keccakf(md->sha3.s);
211 
212  /* store sha3.s[] as little-endian bytes into sha3.sb */
213  for(i = 0; i < SHA3_KECCAK_SPONGE_WORDS; i++) {
214  STORE64L(md->sha3.s[i], md->sha3.sb + i * 8);
215  }
216 
217  XMEMCPY(hash, md->sha3.sb, md->sha3.capacity_words * 4);
218  return CRYPT_OK;
219 }

266 int sha3_process(hash_state *md, const unsigned char *in, unsigned long inlen)
267 {
268  /* 0...7 -- how much is needed to have a word */
269  unsigned old_tail = (8 - md->sha3.byte_index) & 7;
270 
271  unsigned long words;
272  unsigned tail;
273  unsigned long i;
274 
275  if (inlen == 0) return CRYPT_OK; /* nothing to do */
276  LTC_ARGCHK(md != NULL);
277  LTC_ARGCHK(in != NULL);
278 
279  if(inlen < old_tail) { /* have no complete word or haven't started the word yet */
280  while (inlen--) md->sha3.saved |= (ulong64) (*(in++)) << ((md->sha3.byte_index++) * 8);
281  return CRYPT_OK;
282  }
283 
284  if(old_tail) { /* will have one word to process */
285  inlen -= old_tail;
286  while (old_tail--) md->sha3.saved |= (ulong64) (*(in++)) << ((md->sha3.byte_index++) * 8);
287  /* now ready to add saved to the sponge */
288  md->sha3.s[md->sha3.word_index] ^= md->sha3.saved;
289  md->sha3.byte_index = 0;
290  md->sha3.saved = 0;
291  if(++md->sha3.word_index == (SHA3_KECCAK_SPONGE_WORDS - md->sha3.capacity_words)) {
292  keccakf(md->sha3.s);
293  md->sha3.word_index = 0;
294  }
295  }
296 
297  /* now work in full words directly from input */
298  words = inlen / sizeof(ulong64);
299  tail = inlen - words * sizeof(ulong64);
300 
301  for(i = 0; i < words; i++, in += sizeof(ulong64)) {
302  ulong64 t;
303  LOAD64L(t, in);
304  md->sha3.s[md->sha3.word_index] ^= t;
305  if(++md->sha3.word_index == (SHA3_KECCAK_SPONGE_WORDS - md->sha3.capacity_words)) {
306  keccakf(md->sha3.s);
307  md->sha3.word_index = 0;
308  }
309  }
310 
311  /* finally, save the partial word */
312  while (tail--) {
313  md->sha3.saved |= (ulong64) (*(in++)) << ((md->sha3.byte_index++) * 8);
314  }
315  return CRYPT_OK;
316 }

333 int sha3_shake_done(hash_state *md, unsigned char *out, unsigned long outlen)
334 {
335  /* IMPORTANT NOTE: sha3_shake_done can be called many times */
336  unsigned long idx;
337  unsigned i;
338 
339  if (outlen == 0) return CRYPT_OK; /* nothing to do */
340  LTC_ARGCHK(md != NULL);
341  LTC_ARGCHK(out != NULL);
342 
343  if (!md->sha3.xof_flag) {
344  /* shake_xof operation must be done only once */
345  md->sha3.s[md->sha3.word_index] ^= (md->sha3.saved ^ (CONST64(0x1F) << (md->sha3.byte_index * 8)));
346  md->sha3.s[SHA3_KECCAK_SPONGE_WORDS - md->sha3.capacity_words - 1] ^= CONST64(0x8000000000000000);
347  keccakf(md->sha3.s);
348  /* store sha3.s[] as little-endian bytes into sha3.sb */
349  for(i = 0; i < SHA3_KECCAK_SPONGE_WORDS; i++) {
350  STORE64L(md->sha3.s[i], md->sha3.sb + i * 8);
351  }
352  md->sha3.byte_index = 0;
353  md->sha3.xof_flag = 1;
354  }
355 
356  for (idx = 0; idx < outlen; idx++) {
357  if(md->sha3.byte_index >= (SHA3_KECCAK_SPONGE_WORDS - md->sha3.capacity_words) * 8) {
358  keccakf(md->sha3.s);
359  /* store sha3.s[] as little-endian bytes into sha3.sb */
360  for(i = 0; i < SHA3_KECCAK_SPONGE_WORDS; i++) {
361  STORE64L(md->sha3.s[i], md->sha3.sb + i * 8);
362  }
363  md->sha3.byte_index = 0;
364  }
365  out[idx] = md->sha3.sb[md->sha3.byte_index++];
366  }
367  return CRYPT_OK;
368 }
