39 #define kTHETA(a, b, c, d) \
40  temp = a^c; temp = temp ^ ROLc(temp, 8) ^ RORc(temp, 8); \
41  b ^= temp; d ^= temp; \
42  temp = b^d; temp = temp ^ ROLc(temp, 8) ^ RORc(temp, 8); \
43  a ^= temp; c ^= temp;

45 #define THETA(k, a, b, c, d) \
46  temp = a^c; temp = temp ^ ROLc(temp, 8) ^ RORc(temp, 8); \
47  b ^= temp ^ k[1]; d ^= temp ^ k[3]; \
48  temp = b^d; temp = temp ^ ROLc(temp, 8) ^ RORc(temp, 8); \
49  a ^= temp ^ k[0]; c ^= temp ^ k[2];

51 #define GAMMA(a, b, c, d) \
52  b ^= ~(d|c); \
53  a ^= c&b; \
54  temp = d; d = a; a = temp;\
55  c ^= a ^ b ^ d; \
56  b ^= ~(d|c); \
