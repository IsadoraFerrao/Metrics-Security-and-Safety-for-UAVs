127 /**
128  Add entropy to the PRNG state
129  @param in The data to add
130  @param inlen Length of the data to add
131  @param prng PRNG state to update
132  @return CRYPT_OK if successful
133 */
134 int yarrow_add_entropy(const unsigned char *in, unsigned long inlen, prng_state *prng)
135 {
136  hash_state md;
137  int err;
138 
139  LTC_ARGCHK(prng != NULL);
140  LTC_ARGCHK(in != NULL);
141  LTC_ARGCHK(inlen > 0);
142 
143  LTC_MUTEX_LOCK(&prng->lock);
144 
145  if ((err = hash_is_valid(prng->u.yarrow.hash)) != CRYPT_OK) {
146  goto LBL_UNLOCK;
147  }
148 
149  /* start the hash */
150  if ((err = hash_descriptor[prng->u.yarrow.hash].init(&md)) != CRYPT_OK) {
151  goto LBL_UNLOCK;
152  }
153 
154  /* hash the current pool */
155  if ((err = hash_descriptor[prng->u.yarrow.hash].process(&md, prng->u.yarrow.pool,
156  hash_descriptor[prng->u.yarrow.hash].hashsize)) != CRYPT_OK) {
157  goto LBL_UNLOCK;
158  }
159 
160  /* add the new entropy */
161  if ((err = hash_descriptor[prng->u.yarrow.hash].process(&md, in, inlen)) != CRYPT_OK) {
162  goto LBL_UNLOCK;
163  }
164 
165  /* store result */
166  err = hash_descriptor[prng->u.yarrow.hash].done(&md, prng->u.yarrow.pool);
167 
168 LBL_UNLOCK:
169  LTC_MUTEX_UNLOCK(&prng->lock);
170  return err;
171 }

173 /**
174  Make the PRNG ready to read from
175  @param prng The PRNG to make active
176  @return CRYPT_OK if successful
177 */
178 int yarrow_ready(prng_state *prng)
179 {
180  int ks, err;
181 
182  LTC_ARGCHK(prng != NULL);
183 
184  LTC_MUTEX_LOCK(&prng->lock);
185 
186  if ((err = hash_is_valid(prng->u.yarrow.hash)) != CRYPT_OK) {
187  goto LBL_UNLOCK;
188  }
189 
190  if ((err = cipher_is_valid(prng->u.yarrow.cipher)) != CRYPT_OK) {
191  goto LBL_UNLOCK;
192  }
193 
194  /* setup CTR mode using the "pool" as the key */
195  ks = (int)hash_descriptor[prng->u.yarrow.hash].hashsize;
196  if ((err = cipher_descriptor[prng->u.yarrow.cipher].keysize(&ks)) != CRYPT_OK) {
197  goto LBL_UNLOCK;
198  }
199 
200  if ((err = ctr_start(prng->u.yarrow.cipher, /* what cipher to use */
201  prng->u.yarrow.pool, /* IV */
202  prng->u.yarrow.pool, ks, /* KEY and key size */
203  0, /* number of rounds */
204  CTR_COUNTER_LITTLE_ENDIAN, /* little endian counter */
205  &prng->u.yarrow.ctr)) != CRYPT_OK) {
206  goto LBL_UNLOCK;
207  }
208  prng->ready = 1;
209 
210 LBL_UNLOCK:
211  LTC_MUTEX_UNLOCK(&prng->lock);
212  return err;
213 }

215 /**
216  Read from the PRNG
217  @param out Destination
218  @param outlen Length of output
219  @param prng The active PRNG to read from
220  @return Number of octets read
221 */
222 unsigned long yarrow_read(unsigned char *out, unsigned long outlen, prng_state *prng)
223 {
224  if (outlen == 0 || prng == NULL || out == NULL) return 0;
225 
226  LTC_MUTEX_LOCK(&prng->lock);
227 
228  if (!prng->ready) {
229  outlen = 0;
230  goto LBL_UNLOCK;
231  }
232 
233  /* put out in predictable state first */
234  zeromem(out, outlen);
235 
236  /* now randomize it */
237  if (ctr_encrypt(out, out, outlen, &prng->u.yarrow.ctr) != CRYPT_OK) {
238  outlen = 0;
239  }
240 
241 LBL_UNLOCK:
242  LTC_MUTEX_UNLOCK(&prng->lock);
243  return outlen;
244 }
