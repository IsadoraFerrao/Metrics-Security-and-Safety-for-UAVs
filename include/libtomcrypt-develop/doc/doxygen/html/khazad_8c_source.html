582  /**
583  Initialize the Khazad block cipher
584  @param key The symmetric key you wish to pass
585  @param keylen The key length in bytes
586  @param num_rounds The number of rounds desired (0 for default)
587  @param skey The key in as scheduled by this function.
588  @return CRYPT_OK if successful
589  */
590 int khazad_setup(const unsigned char *key, int keylen, int num_rounds, symmetric_key *skey)
591 {
592  int r;
593  const ulong64 *S;
594  ulong64 K2, K1;
595 
596  LTC_ARGCHK(key != NULL);
597  LTC_ARGCHK(skey != NULL);
598  if (keylen != 16) {
599  return CRYPT_INVALID_KEYSIZE;
600  }
601  if (num_rounds != 8 && num_rounds != 0) {
602  return CRYPT_INVALID_ROUNDS;
603  }
604 
605  /* use 7th table */
606  S = T7;
607 
608  /*
609  * map unsigned char array cipher key to initial key state (mu):
610  */
611  K2 =
612  ((ulong64)key[ 0] << 56) ^
613  ((ulong64)key[ 1] << 48) ^
614  ((ulong64)key[ 2] << 40) ^
615  ((ulong64)key[ 3] << 32) ^
616  ((ulong64)key[ 4] << 24) ^
617  ((ulong64)key[ 5] << 16) ^
618  ((ulong64)key[ 6] << 8) ^
619  ((ulong64)key[ 7] );
620  K1 =
621  ((ulong64)key[ 8] << 56) ^
622  ((ulong64)key[ 9] << 48) ^
623  ((ulong64)key[10] << 40) ^
624  ((ulong64)key[11] << 32) ^
625  ((ulong64)key[12] << 24) ^
626  ((ulong64)key[13] << 16) ^
627  ((ulong64)key[14] << 8) ^
628  ((ulong64)key[15] );
629 
630  /*
631  * compute the round keys:
632  */
633  for (r = 0; r <= R; r++) {
634  /*
635  * K[r] = rho(c[r], K1) ^ K2;
636  */
637  skey->khazad.roundKeyEnc[r] =
638  T0[(int)(K1 >> 56) ] ^
639  T1[(int)(K1 >> 48) & 0xff] ^
640  T2[(int)(K1 >> 40) & 0xff] ^
641  T3[(int)(K1 >> 32) & 0xff] ^
642  T4[(int)(K1 >> 24) & 0xff] ^
643  T5[(int)(K1 >> 16) & 0xff] ^
644  T6[(int)(K1 >> 8) & 0xff] ^
645  T7[(int)(K1 ) & 0xff] ^
646  c[r] ^ K2;
647  K2 = K1; K1 = skey->khazad.roundKeyEnc[r];
648  }
649  /*
650  * compute the inverse key schedule:
651  * K'^0 = K^R, K'^R = K^0, K'^r = theta(K^{R-r})
652  */
653  skey->khazad.roundKeyDec[0] = skey->khazad.roundKeyEnc[R];
654  for (r = 1; r < R; r++) {
655  K1 = skey->khazad.roundKeyEnc[R - r];
656  skey->khazad.roundKeyDec[r] =
657  T0[(int)S[(int)(K1 >> 56) ] & 0xff] ^
658  T1[(int)S[(int)(K1 >> 48) & 0xff] & 0xff] ^
659  T2[(int)S[(int)(K1 >> 40) & 0xff] & 0xff] ^
660  T3[(int)S[(int)(K1 >> 32) & 0xff] & 0xff] ^
661  T4[(int)S[(int)(K1 >> 24) & 0xff] & 0xff] ^
662  T5[(int)S[(int)(K1 >> 16) & 0xff] & 0xff] ^
663  T6[(int)S[(int)(K1 >> 8) & 0xff] & 0xff] ^
664  T7[(int)S[(int)(K1 ) & 0xff] & 0xff];
665  }
666  skey->khazad.roundKeyDec[R] = skey->khazad.roundKeyEnc[0];
667 
668  return CRYPT_OK;
669 }

671 static void khazad_crypt(const unsigned char *plaintext, unsigned char *ciphertext,
672  const ulong64 *roundKey) {
673  int r;
674  ulong64 state;
675  /*
676  * map plaintext block to cipher state (mu)
677  * and add initial round key (sigma[K^0]):
678  */
679  state =
680  ((ulong64)plaintext[0] << 56) ^
681  ((ulong64)plaintext[1] << 48) ^
682  ((ulong64)plaintext[2] << 40) ^
683  ((ulong64)plaintext[3] << 32) ^
684  ((ulong64)plaintext[4] << 24) ^
685  ((ulong64)plaintext[5] << 16) ^
686  ((ulong64)plaintext[6] << 8) ^
687  ((ulong64)plaintext[7] ) ^
688  roundKey[0];
689 
690  /*
691  * R - 1 full rounds:
692  */
693  for (r = 1; r < R; r++) {
694  state =
695  T0[(int)(state >> 56) ] ^
696  T1[(int)(state >> 48) & 0xff] ^
697  T2[(int)(state >> 40) & 0xff] ^
698  T3[(int)(state >> 32) & 0xff] ^
699  T4[(int)(state >> 24) & 0xff] ^
700  T5[(int)(state >> 16) & 0xff] ^
701  T6[(int)(state >> 8) & 0xff] ^
702  T7[(int)(state ) & 0xff] ^
703  roundKey[r];
704  }
705 
706  /*
707  * last round:
708  */
709  state =
710  (T0[(int)(state >> 56) ] & CONST64(0xff00000000000000)) ^
711  (T1[(int)(state >> 48) & 0xff] & CONST64(0x00ff000000000000)) ^
712  (T2[(int)(state >> 40) & 0xff] & CONST64(0x0000ff0000000000)) ^
713  (T3[(int)(state >> 32) & 0xff] & CONST64(0x000000ff00000000)) ^
714  (T4[(int)(state >> 24) & 0xff] & CONST64(0x00000000ff000000)) ^
715  (T5[(int)(state >> 16) & 0xff] & CONST64(0x0000000000ff0000)) ^
716  (T6[(int)(state >> 8) & 0xff] & CONST64(0x000000000000ff00)) ^
717  (T7[(int)(state ) & 0xff] & CONST64(0x00000000000000ff)) ^
718  roundKey[R];
719 
720  /*
721  * map cipher state to ciphertext block (mu^{-1}):
722  */
723  ciphertext[0] = (unsigned char)(state >> 56);
724  ciphertext[1] = (unsigned char)(state >> 48);
725  ciphertext[2] = (unsigned char)(state >> 40);
726  ciphertext[3] = (unsigned char)(state >> 32);
727  ciphertext[4] = (unsigned char)(state >> 24);
728  ciphertext[5] = (unsigned char)(state >> 16);
729  ciphertext[6] = (unsigned char)(state >> 8);
730  ciphertext[7] = (unsigned char)(state );
731 }
