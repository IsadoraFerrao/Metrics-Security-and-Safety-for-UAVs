574 /* find a hole and free as required, return -1 if no hole found */
575 static int _find_hole(void)
576 {
577  unsigned x;
578  int y, z;
579  for (z = -1, y = INT_MAX, x = 0; x < FP_ENTRIES; x++) {
580  if (fp_cache[x].lru_count < y && fp_cache[x].lock == 0) {
581  z = x;
582  y = fp_cache[x].lru_count;
583  }
584  }
585 
586  /* decrease all */
587  for (x = 0; x < FP_ENTRIES; x++) {
588  if (fp_cache[x].lru_count > 3) {
589  --(fp_cache[x].lru_count);
590  }
591  }
592 
593  /* free entry z */
594  if (z >= 0 && fp_cache[z].g) {
595  if (fp_cache[z].mu != NULL) {
596  mp_clear(fp_cache[z].mu);
597  fp_cache[z].mu = NULL;
598  }
599  ltc_ecc_del_point(fp_cache[z].g);
600  fp_cache[z].g = NULL;
601  for (x = 0; x < (1U<<FP_LUT); x++) {
602  ltc_ecc_del_point(fp_cache[z].LUT[x]);
603  fp_cache[z].LUT[x] = NULL;
604  }
605  fp_cache[z].lru_count = 0;
606  }
607  return z;
608 }

666 /* build the LUT by spacing the bits of the input by #modulus/FP_LUT bits apart
667  *
668  * The algorithm builds patterns in increasing bit order by first making all
669  * single bit input patterns, then all two bit input patterns and so on
670  */
671 static int _build_lut(int idx, void *a, void *modulus, void *mp, void *mu)
672 {
673  unsigned x, y, err, bitlen, lut_gap;
674  void *tmp;
675 
676  tmp = NULL;
677 
678  /* sanity check to make sure lut_order table is of correct size, should compile out to a NOP if true */
679  if ((sizeof(lut_orders) / sizeof(lut_orders[0])) < (1U<<FP_LUT)) {
680  err = CRYPT_INVALID_ARG;
681  goto DONE;
682  }
683 
684  /* get bitlen and round up to next multiple of FP_LUT */
685  bitlen = mp_unsigned_bin_size(modulus) << 3;
686  x = bitlen % FP_LUT;
687  if (x) {
688  bitlen += FP_LUT - x;
689  }
690  lut_gap = bitlen / FP_LUT;
691 
692  /* init the mu */
693  if ((err = mp_init_copy(&fp_cache[idx].mu, mu)) != CRYPT_OK) {
694  goto ERR;
695  }
696 
697  /* copy base */
698  if ((mp_mulmod(fp_cache[idx].g->x, mu, modulus, fp_cache[idx].LUT[1]->x) != CRYPT_OK) ||
699  (mp_mulmod(fp_cache[idx].g->y, mu, modulus, fp_cache[idx].LUT[1]->y) != CRYPT_OK) ||
700  (mp_mulmod(fp_cache[idx].g->z, mu, modulus, fp_cache[idx].LUT[1]->z) != CRYPT_OK)) { goto ERR; }
701 
702  /* make all single bit entries */
703  for (x = 1; x < FP_LUT; x++) {
704  if ((mp_copy(fp_cache[idx].LUT[1<<(x-1)]->x, fp_cache[idx].LUT[1<<x]->x) != CRYPT_OK) ||
705  (mp_copy(fp_cache[idx].LUT[1<<(x-1)]->y, fp_cache[idx].LUT[1<<x]->y) != CRYPT_OK) ||
706  (mp_copy(fp_cache[idx].LUT[1<<(x-1)]->z, fp_cache[idx].LUT[1<<x]->z) != CRYPT_OK)) { goto ERR; }
707 
708  /* now double it bitlen/FP_LUT times */
709  for (y = 0; y < lut_gap; y++) {
710  if ((err = ltc_mp.ecc_ptdbl(fp_cache[idx].LUT[1<<x], fp_cache[idx].LUT[1<<x], a, modulus, mp)) != CRYPT_OK) {
711  goto ERR;
712  }
713  }
714  }
715 
716  /* now make all entries in increase order of hamming weight */
717  for (x = 2; x <= FP_LUT; x++) {
718  for (y = 0; y < (1UL<<FP_LUT); y++) {
719  if (lut_orders[y].ham != (int)x) continue;
720 
721  /* perform the add */
722  if ((err = ltc_mp.ecc_ptadd(fp_cache[idx].LUT[lut_orders[y].terma], fp_cache[idx].LUT[lut_orders[y].termb],
723  fp_cache[idx].LUT[y], a, modulus, mp)) != CRYPT_OK) {
724  goto ERR;
725  }
726  }
727  }
728 
729  /* now map all entries back to affine space to make point addition faster */
730  if ((err = mp_init(&tmp)) != CRYPT_OK) { goto ERR; }
731  for (x = 1; x < (1UL<<FP_LUT); x++) {
732  /* convert z to normal from montgomery */
733  if ((err = mp_montgomery_reduce(fp_cache[idx].LUT[x]->z, modulus, mp)) != CRYPT_OK) { goto ERR; }
734 
735  /* invert it */
736  if ((err = mp_invmod(fp_cache[idx].LUT[x]->z, modulus, fp_cache[idx].LUT[x]->z)) != CRYPT_OK) { goto ERR; }
737 
738  /* now square it */
739  if ((err = mp_sqrmod(fp_cache[idx].LUT[x]->z, modulus, tmp)) != CRYPT_OK) { goto ERR; }
740 
741  /* fix x */
742  if ((err = mp_mulmod(fp_cache[idx].LUT[x]->x, tmp, modulus, fp_cache[idx].LUT[x]->x)) != CRYPT_OK) { goto ERR; }
743 
744  /* get 1/z^3 */
745  if ((err = mp_mulmod(tmp, fp_cache[idx].LUT[x]->z, modulus, tmp)) != CRYPT_OK) { goto ERR; }
746 
747  /* fix y */
748  if ((err = mp_mulmod(fp_cache[idx].LUT[x]->y, tmp, modulus, fp_cache[idx].LUT[x]->y)) != CRYPT_OK) { goto ERR; }
749 
750  /* free z */
751  mp_clear(fp_cache[idx].LUT[x]->z);
752  fp_cache[idx].LUT[x]->z = NULL;
753  }
754  mp_clear(tmp);
755 
756  return CRYPT_OK;
757 ERR:
758  err = CRYPT_MEM;
759 DONE:
760  for (y = 0; y < (1U<<FP_LUT); y++) {
761  ltc_ecc_del_point(fp_cache[idx].LUT[y]);
762  fp_cache[idx].LUT[y] = NULL;
763  }
764  ltc_ecc_del_point(fp_cache[idx].g);
765  fp_cache[idx].g = NULL;
766  fp_cache[idx].lru_count = 0;
767  if (fp_cache[idx].mu != NULL) {
768  mp_clear(fp_cache[idx].mu);
769  fp_cache[idx].mu = NULL;
770  }
771  if (tmp != NULL) {
772  mp_clear(tmp);
773  }
774  return err;
775 }

777 /* perform a fixed point ECC mulmod */
778 static int _accel_fp_mul(int idx, void *k, ecc_point *R, void *a, void *modulus, void *mp, int map)
779 {
780  unsigned char kb[128];
781  int x;
782  unsigned y, z, err, bitlen, bitpos, lut_gap, first;
783  void *tk, *order;
784 
785  /* if it's smaller than modulus we fine */
786  if (mp_unsigned_bin_size(k) > mp_unsigned_bin_size(modulus)) {
787  /* find order */
788  y = mp_unsigned_bin_size(modulus);
789  for (x = 0; ltc_ecc_sets[x].size; x++) {
790  if (y <= (unsigned)ltc_ecc_sets[x].size) break;
791  }
792 
793  /* back off if we are on the 521 bit curve */
794  if (y == 66) --x;
795 
796  if ((err = mp_init(&order)) != CRYPT_OK) {
797  return err;
798  }
799  if ((err = mp_read_radix(order, ltc_ecc_sets[x].order, 16)) != CRYPT_OK) {
800  mp_clear(&order);
801  return err;
802  }
803 
804  /* k must be less than modulus */
805  if (mp_cmp(k, order) != LTC_MP_LT) {
806  if ((err = mp_init(&tk)) != CRYPT_OK) {
807  mp_clear(order);
808  return err;
809  }
810  if ((err = mp_mod(k, order, tk)) != CRYPT_OK) {
811  mp_clear(tk);
812  mp_clear(order);
813  return err;
814  }
815  } else {
816  tk = k;
817  }
818  mp_clear(order);
819  } else {
820  tk = k;
821  }
822 
823  /* get bitlen and round up to next multiple of FP_LUT */
824  bitlen = mp_unsigned_bin_size(modulus) << 3;
825  x = bitlen % FP_LUT;
826  if (x) {
827  bitlen += FP_LUT - x;
828  }
829  lut_gap = bitlen / FP_LUT;
830 
831  /* get the k value */
832  if (mp_unsigned_bin_size(tk) > (sizeof(kb) - 2)) {
833  if (tk != k) {
834  mp_clear(tk);
835  }
836  return CRYPT_BUFFER_OVERFLOW;
837  }
838 
839  /* store k */
840  zeromem(kb, sizeof(kb));
841  if ((err = mp_to_unsigned_bin(tk, kb)) != CRYPT_OK) {
842  if (tk != k) {
843  mp_clear(tk);
844  }
845  return err;
846  }
847 
848  /* let's reverse kb so it's little endian */
849  x = 0;
850  y = mp_unsigned_bin_size(tk) - 1;
851  if (tk != k) {
852  mp_clear(tk);
853  }
854  while ((unsigned)x < y) {
855  z = kb[x]; kb[x] = kb[y]; kb[y] = z;
856  ++x; --y;
857  }
858 
859  /* at this point we can start, yipee */
860  first = 1;
861  for (x = lut_gap-1; x >= 0; x--) {
862  /* extract FP_LUT bits from kb spread out by lut_gap bits and offset by x bits from the start */
863  bitpos = x;
864  for (y = z = 0; y < FP_LUT; y++) {
865  z |= ((kb[bitpos>>3] >> (bitpos&7)) & 1) << y;
866  bitpos += lut_gap; /* it's y*lut_gap + x, but here we can avoid the mult in each loop */
867  }
868 
869  /* double if not first */
870  if (!first) {
871  if ((err = ltc_mp.ecc_ptdbl(R, R, a, modulus, mp)) != CRYPT_OK) {
872  return err;
873  }
874  }
875 
876  /* add if not first, otherwise copy */
877  if (!first && z) {
878  if ((err = ltc_mp.ecc_ptadd(R, fp_cache[idx].LUT[z], R, a, modulus, mp)) != CRYPT_OK) {
879  return err;
880  }
881  } else if (z) {
882  if ((mp_copy(fp_cache[idx].LUT[z]->x, R->x) != CRYPT_OK) ||
883  (mp_copy(fp_cache[idx].LUT[z]->y, R->y) != CRYPT_OK) ||
884  (mp_copy(fp_cache[idx].mu, R->z) != CRYPT_OK)) { return CRYPT_MEM; }
885  first = 0;
886  }
887  }
888  z = 0;
889  zeromem(kb, sizeof(kb));
890  /* map R back from projective space */
891  if (map) {
892  err = ltc_ecc_map(R, modulus, mp);
893  } else {
894  err = CRYPT_OK;
895  }
896  return err;
897 }

936  mp_clear(tka);
937  mp_clear(order);
938  return err;
939  }
940  } else {
941  tka = kA;
942  }
943  mp_clear(order);
944  } else {
945  tka = kA;
946  }
947 
948  /* if it's smaller than modulus we fine */
949  if (mp_unsigned_bin_size(kB) > mp_unsigned_bin_size(modulus)) {
950  /* find order */
951  y = mp_unsigned_bin_size(modulus);
952  for (x = 0; ltc_ecc_sets[x].size; x++) {
953  if (y <= (unsigned)ltc_ecc_sets[x].size) break;
954  }

956  /* back off if we are on the 521 bit curve */
957  if (y == 66) --x;
958

960  return err;
961  }
962  if ((err = mp_read_radix(order, ltc_ecc_sets[x].order, 16)) != CRYPT_OK) {
963  mp_clear(&order);
964  return err;
965  }

968  if (mp_cmp(kB, order) != LTC_MP_LT) {

970  mp_clear(order);

972  }

974  mp_clear(tkb);
975  mp_clear(order);
976  return err;
977  }
978  } else {

980  }
981  mp_clear(order);
982  } else {
983  tkb = kB;
984  }
985

987  bitlen = mp_unsigned_bin_size(modulus) << 3;
988  x = bitlen % FP_LUT;
989  if (x) {
990  bitlen += FP_LUT - x;
991  }
992  lut_gap = bitlen / FP_LUT;
993 
994  /* get the k value */
995  if ((mp_unsigned_bin_size(tka) > (sizeof(kb[0]) - 2)) || (mp_unsigned_bin_size(tkb) > (sizeof(kb[0]) - 2)) ) {
996  if (tka != kA) {

998  }
999  if (tkb != kB) {

1001  }
1002  return CRYPT_BUFFER_OVERFLOW;
1003  }

1007  if ((err = mp_to_unsigned_bin(tka, kb[0])) != CRYPT_OK) {
1008  if (tka != kA) {
1009  mp_clear(tka);
1010  }
1011  if (tkb != kB) {
1012  mp_clear(tkb);
1013  }
1014  return err;
1015  }

1018  x = 0;
1019  y = mp_unsigned_bin_size(tka) - 1;

1021  mp_clear(tka);
1022  }

1025  ++x; --y;
1026  }
1027

1029  if ((err = mp_to_unsigned_bin(tkb, kb[1])) != CRYPT_OK) {
1030  if (tkb != kB) {
1031  mp_clear(tkb);
1032  }
1033  return err;
1034  }

1037  y = mp_unsigned_bin_size(tkb) - 1;
1038  if (tkb != kB) {
1039  mp_clear(tkb);
1040  }
1041  while ((unsigned)x < y) {
1042  z = kb[1][x]; kb[1][x] = kb[1][y]; kb[1][y] = z;
1043  ++x; --y;
1044  }
1045 
1046  /* at this point we can start, yipee */
1047  first = 1;
1048  for (x = lut_gap-1; x >= 0; x--) {
1049  /* extract FP_LUT bits from kb spread out by lut_gap bits and offset by x bits from the start */
1050  bitpos = x;
1051  for (y = zA = zB = 0; y < FP_LUT; y++) {
1052  zA |= ((kb[0][bitpos>>3] >> (bitpos&7)) & 1) << y;
1053  zB |= ((kb[1][bitpos>>3] >> (bitpos&7)) & 1) << y;
1054  bitpos += lut_gap; /* it's y*lut_gap + x, but here we can avoid the mult in each loop */
1055  }
1056 
1057  /* double if not first */
1058  if (!first) {
1059  if ((err = ltc_mp.ecc_ptdbl(R, R, a, modulus, mp)) != CRYPT_OK) {
1060  return err;
1061  }
1062  }
1063 
1064  /* add if not first, otherwise copy */
1065  if (!first) {
1066  if (zA) {
1067  if ((err = ltc_mp.ecc_ptadd(R, fp_cache[idx1].LUT[zA], R, a, modulus, mp)) != CRYPT_OK) {
1068  return err;
1069  }
1070  }
1071  if (zB) {
1072  if ((err = ltc_mp.ecc_ptadd(R, fp_cache[idx2].LUT[zB], R, a, modulus, mp)) != CRYPT_OK) {
1073  return err;
1074  }
1075  }
1076  } else {
1077  if (zA) {
1078  if ((mp_copy(fp_cache[idx1].LUT[zA]->x, R->x) != CRYPT_OK) ||
1079  (mp_copy(fp_cache[idx1].LUT[zA]->y, R->y) != CRYPT_OK) ||
1080  (mp_copy(fp_cache[idx1].mu, R->z) != CRYPT_OK)) { return CRYPT_MEM; }
1081  first = 0;
1082  }
1083  if (zB && first == 0) {
1084  if (zB) {
1085  if ((err = ltc_mp.ecc_ptadd(R, fp_cache[idx2].LUT[zB], R, a, modulus, mp)) != CRYPT_OK) {
1086  return err;
1087  }
1088  }
1089  } else if (zB && first == 1) {
1090  if ((mp_copy(fp_cache[idx2].LUT[zB]->x, R->x) != CRYPT_OK) ||
1091  (mp_copy(fp_cache[idx2].LUT[zB]->y, R->y) != CRYPT_OK) ||
1092  (mp_copy(fp_cache[idx2].mu, R->z) != CRYPT_OK)) { return CRYPT_MEM; }
1093  first = 0;
1094  }
1095  }
1096  }
1097  zeromem(kb, sizeof(kb));
1098  return ltc_ecc_map(R, modulus, mp);
1099 }

1471  goto save_err;
