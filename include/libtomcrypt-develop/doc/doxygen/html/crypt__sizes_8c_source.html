293 /* crypt_get_size()
294  * sizeout will be the size (bytes) of the named struct or union
295  * return -1 if named item not found
296  */
297 int crypt_get_size(const char* namein, unsigned int *sizeout) {
298  int i;
299  int count = sizeof(_crypt_sizes) / sizeof(_crypt_sizes[0]);
300  for (i=0; i<count; i++) {
301  if (XSTRCMP(_crypt_sizes[i].name, namein) == 0) {
302  *sizeout = _crypt_sizes[i].size;
303  return 0;
304  }
305  }
306  return -1;
307 }

309 /* crypt_list_all_sizes()
310  * if names_list is NULL, names_list_size will be the minimum
311  * size needed to receive the complete names_list
312  * if names_list is NOT NULL, names_list must be the addr with
313  * sufficient memory allocated into which the names_list
314  * is to be written. Also, the value in names_list_size
315  * sets the upper bound of the number of characters to be
316  * written.
317  * a -1 return value signifies insufficient space made available
318  */
319 int crypt_list_all_sizes(char *names_list, unsigned int *names_list_size) {
320  int i;
321  unsigned int total_len = 0;
322  char *ptr;
323  int number_len;
324  int count = sizeof(_crypt_sizes) / sizeof(_crypt_sizes[0]);
325 
326  /* calculate amount of memory required for the list */
327  for (i=0; i<count; i++) {
328  number_len = snprintf(NULL, 0, "%s,%u\n", _crypt_sizes[i].name, _crypt_sizes[i].size);
329  if (number_len < 0) {
330  return -1;
331  }
332  total_len += number_len;
333  /* this last +1 is for newlines (and ending NULL) */
334  }
335 
336  if (names_list == NULL) {
337  *names_list_size = total_len;
338  } else {
339  if (total_len > *names_list_size) {
340  return -1;
341  }
342  /* build the names list */
343  ptr = names_list;
344  for (i=0; i<count; i++) {
345  number_len = snprintf(ptr, total_len, "%s,%u\n", _crypt_sizes[i].name, _crypt_sizes[i].size);
346  if (number_len < 0) return -1;
347  if ((unsigned int)number_len > total_len) return -1;
348  total_len -= number_len;
349  ptr += number_len;
350  }
351  /* to remove the trailing new-line */
352  ptr -= 1;
353  *ptr = 0;
354  }
355  return 0;
356 }
