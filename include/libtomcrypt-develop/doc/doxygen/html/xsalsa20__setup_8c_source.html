29 /* use modified salsa20 doubleround (no final addition as in salsa20) */
30 static void _xsalsa20_doubleround(ulong32 *x, int rounds)
31 {
32  int i;
33 
34  for (i = rounds; i > 0; i -= 2) {
35  /* columnround */
36  QUARTERROUND( 0, 4, 8,12)
37  QUARTERROUND( 5, 9,13, 1)
38  QUARTERROUND(10,14, 2, 6)
39  QUARTERROUND(15, 3, 7,11)
40  /* rowround */
41  QUARTERROUND( 0, 1, 2, 3)
42  QUARTERROUND( 5, 6, 7, 4)
43  QUARTERROUND(10,11, 8, 9)
44  QUARTERROUND(15,12,13,14)
45  }
46 }
