15  * works in increments of 8-bit bytes. Keys must be submitted as whole bytes
16  * and shorter keys will be right-null-padded to 16 bytes. Likewise, an iv
17  * may be any length up to 8 bytes and will be padded out to 8 bytes.
18  *

20  * ECRYPT_process_blocks() and ECRYPT_process_bytes(). That version allowed

76 /* Square a 32-bit unsigned integer to obtain the 64-bit result and return */
77 /* the upper 32 bits XOR the lower 32 bits */
78 static LTC_INLINE ulong32 _rabbit_g_func(ulong32 x)
79 {
80  ulong32 a, b, h, l;
81 
82  /* Construct high and low argument for squaring */
83  a = x & 0xFFFF;
84  b = x >> 16;
85 
86  /* Calculate high and low result of squaring */
87  h = ((((ulong32)(a*a)>>17) + (ulong32)(a*b))>>15) + b*b;
88  l = x * x;
89 
90  /* Return high XOR low */
91  return (ulong32)(h^l);
92 }

96 /* Calculate the next internal state */
97 static LTC_INLINE void _rabbit_next_state(rabbit_ctx *p_instance)
98 {
99  ulong32 g[8], c_old[8], i;
100 
101  /* Save old counter values */
102  for (i=0; i<8; i++) {
103  c_old[i] = p_instance->c[i];
104  }
105 
106  /* Calculate new counter values */
107  p_instance->c[0] = (ulong32)(p_instance->c[0] + 0x4D34D34D + p_instance->carry);
108  p_instance->c[1] = (ulong32)(p_instance->c[1] + 0xD34D34D3 + (p_instance->c[0] < c_old[0]));
109  p_instance->c[2] = (ulong32)(p_instance->c[2] + 0x34D34D34 + (p_instance->c[1] < c_old[1]));
110  p_instance->c[3] = (ulong32)(p_instance->c[3] + 0x4D34D34D + (p_instance->c[2] < c_old[2]));
111  p_instance->c[4] = (ulong32)(p_instance->c[4] + 0xD34D34D3 + (p_instance->c[3] < c_old[3]));
112  p_instance->c[5] = (ulong32)(p_instance->c[5] + 0x34D34D34 + (p_instance->c[4] < c_old[4]));
113  p_instance->c[6] = (ulong32)(p_instance->c[6] + 0x4D34D34D + (p_instance->c[5] < c_old[5]));
114  p_instance->c[7] = (ulong32)(p_instance->c[7] + 0xD34D34D3 + (p_instance->c[6] < c_old[6]));
115  p_instance->carry = (p_instance->c[7] < c_old[7]);
116 
117  /* Calculate the g-values */
118  for (i=0;i<8;i++) {
119  g[i] = _rabbit_g_func((ulong32)(p_instance->x[i] + p_instance->c[i]));
120  }
121 
122  /* Calculate new state values */
123  p_instance->x[0] = (ulong32)(g[0] + ROLc(g[7],16) + ROLc(g[6], 16));
124  p_instance->x[1] = (ulong32)(g[1] + ROLc(g[0], 8) + g[7]);
125  p_instance->x[2] = (ulong32)(g[2] + ROLc(g[1],16) + ROLc(g[0], 16));
126  p_instance->x[3] = (ulong32)(g[3] + ROLc(g[2], 8) + g[1]);
127  p_instance->x[4] = (ulong32)(g[4] + ROLc(g[3],16) + ROLc(g[2], 16));
128  p_instance->x[5] = (ulong32)(g[5] + ROLc(g[4], 8) + g[3]);
129  p_instance->x[6] = (ulong32)(g[6] + ROLc(g[5],16) + ROLc(g[4], 16));
130  p_instance->x[7] = (ulong32)(g[7] + ROLc(g[6], 8) + g[5]);
131 }

135 static LTC_INLINE void _rabbit_gen_1_block(rabbit_state* st, unsigned char *out)
136 {
137  ulong32 *ptr;
138 
139  /* Iterate the work context once */
140  _rabbit_next_state(&(st->work_ctx));
141 
142  /* Generate 16 bytes of pseudo-random data */
143  ptr = (ulong32*)&(st->work_ctx.x);
144  STORE32L((ptr[0] ^ (ptr[5]>>16) ^ (ulong32)(ptr[3]<<16)), out+ 0);
145  STORE32L((ptr[2] ^ (ptr[7]>>16) ^ (ulong32)(ptr[5]<<16)), out+ 4);
146  STORE32L((ptr[4] ^ (ptr[1]>>16) ^ (ulong32)(ptr[7]<<16)), out+ 8);
147  STORE32L((ptr[6] ^ (ptr[3]>>16) ^ (ulong32)(ptr[1]<<16)), out+12);
148 }

152 /* Key setup */
153 int rabbit_setup(rabbit_state* st, const unsigned char *key, unsigned long keylen)
154 {
155  ulong32 k0, k1, k2, k3, i;
156  unsigned char tmpkey[16] = {0};
157 
158  LTC_ARGCHK(st != NULL);
159  LTC_ARGCHK(key != NULL);
160  LTC_ARGCHK(keylen <= 16);
161 
162  /* init state */
163  XMEMSET(st, 0, sizeof(rabbit_state));
164 
165  /* pad key in tmpkey */
166  XMEMCPY(tmpkey, key, keylen);
167 
168  /* Generate four subkeys */
169  LOAD32L(k0, tmpkey+ 0);
170  LOAD32L(k1, tmpkey+ 4);
171  LOAD32L(k2, tmpkey+ 8);
172  LOAD32L(k3, tmpkey+12);
173 
174 #ifdef LTC_CLEAN_STACK
175  /* done with tmpkey, wipe it */
176  zeromem(tmpkey, sizeof(tmpkey));
177 #endif
178 
179  /* Generate initial state variables */
180  st->master_ctx.x[0] = k0;
181  st->master_ctx.x[2] = k1;
182  st->master_ctx.x[4] = k2;
183  st->master_ctx.x[6] = k3;
184  st->master_ctx.x[1] = (ulong32)(k3<<16) | (k2>>16);
185  st->master_ctx.x[3] = (ulong32)(k0<<16) | (k3>>16);
186  st->master_ctx.x[5] = (ulong32)(k1<<16) | (k0>>16);
187  st->master_ctx.x[7] = (ulong32)(k2<<16) | (k1>>16);
188 
189  /* Generate initial counter values */
190  st->master_ctx.c[0] = ROLc(k2, 16);
191  st->master_ctx.c[2] = ROLc(k3, 16);
192  st->master_ctx.c[4] = ROLc(k0, 16);
193  st->master_ctx.c[6] = ROLc(k1, 16);
194  st->master_ctx.c[1] = (k0&0xFFFF0000) | (k1&0xFFFF);
195  st->master_ctx.c[3] = (k1&0xFFFF0000) | (k2&0xFFFF);
196  st->master_ctx.c[5] = (k2&0xFFFF0000) | (k3&0xFFFF);
197  st->master_ctx.c[7] = (k3&0xFFFF0000) | (k0&0xFFFF);
198 
199  /* Clear carry bit */
200  st->master_ctx.carry = 0;
201 
202  /* Iterate the master context four times */
203  for (i=0; i<4; i++) {
204  _rabbit_next_state(&(st->master_ctx));
205  }
206 
207  /* Modify the counters */
208  for (i=0; i<8; i++) {
209  st->master_ctx.c[i] ^= st->master_ctx.x[(i+4)&0x7];
210  }
211 
212  /* Copy master instance to work instance */
213  for (i=0; i<8; i++) {
214  st->work_ctx.x[i] = st->master_ctx.x[i];
215  st->work_ctx.c[i] = st->master_ctx.c[i];
216  }
217  st->work_ctx.carry = st->master_ctx.carry;
218  /* ...and prepare block for crypt() */
219  XMEMSET(&(st->block), 0, sizeof(st->block));
220  st->unused = 0;
221 
222  return CRYPT_OK;
223 }

227 /* IV setup */
228 int rabbit_setiv(rabbit_state* st, const unsigned char *iv, unsigned long ivlen)
229 {
230  ulong32 i0, i1, i2, i3, i;
231  unsigned char tmpiv[8] = {0};
232 
233  LTC_ARGCHK(st != NULL);
234  LTC_ARGCHK(iv != NULL || ivlen == 0);
235  LTC_ARGCHK(ivlen <= 8);
236 
237  /* pad iv in tmpiv */
238  if (iv && ivlen > 0) XMEMCPY(tmpiv, iv, ivlen);
239 
240  /* Generate four subvectors */
241  LOAD32L(i0, tmpiv+0);
242  LOAD32L(i2, tmpiv+4);
243  i1 = (i0>>16) | (i2&0xFFFF0000);
244  i3 = (i2<<16) | (i0&0x0000FFFF);
245 
246  /* Modify counter values */
247  st->work_ctx.c[0] = st->master_ctx.c[0] ^ i0;
248  st->work_ctx.c[1] = st->master_ctx.c[1] ^ i1;
249  st->work_ctx.c[2] = st->master_ctx.c[2] ^ i2;
250  st->work_ctx.c[3] = st->master_ctx.c[3] ^ i3;
251  st->work_ctx.c[4] = st->master_ctx.c[4] ^ i0;
252  st->work_ctx.c[5] = st->master_ctx.c[5] ^ i1;
253  st->work_ctx.c[6] = st->master_ctx.c[6] ^ i2;
254  st->work_ctx.c[7] = st->master_ctx.c[7] ^ i3;
255 
256  /* Copy state variables */
257  for (i=0; i<8; i++) {
258  st->work_ctx.x[i] = st->master_ctx.x[i];
259  }
260  st->work_ctx.carry = st->master_ctx.carry;
261 
262  /* Iterate the work context four times */
263  for (i=0; i<4; i++) {
264  _rabbit_next_state(&(st->work_ctx));
265  }
266 
267  /* reset keystream buffer and unused count */
268  XMEMSET(&(st->block), 0, sizeof(st->block));
269  st->unused = 0;
270 
271  return CRYPT_OK;
272 }

276 /* Crypt a chunk of any size (encrypt/decrypt) */
277 int rabbit_crypt(rabbit_state* st, const unsigned char *in, unsigned long inlen, unsigned char *out)
278 {
279  unsigned char buf[16];
280  unsigned long i, j;
281 
282  if (inlen == 0) return CRYPT_OK; /* nothing to do */
283 
284  LTC_ARGCHK(st != NULL);
285  LTC_ARGCHK(in != NULL);
286  LTC_ARGCHK(out != NULL);
287 
288  if (st->unused > 0) {
289  j = MIN(st->unused, inlen);
290  for (i = 0; i < j; ++i, st->unused--) out[i] = in[i] ^ st->block[16 - st->unused];
291  inlen -= j;
292  if (inlen == 0) return CRYPT_OK;
293  out += j;
294  in += j;
295  }
296  for (;;) {
297  /* gen a block for buf */
298  _rabbit_gen_1_block(st, buf);
299  if (inlen <= 16) {
300  /* XOR and send to out */
301  for (i = 0; i < inlen; ++i) out[i] = in[i] ^ buf[i];
302  st->unused = 16 - inlen;
303  /* copy remainder to block */
304  for (i = inlen; i < 16; ++i) st->block[i] = buf[i];
305  return CRYPT_OK;
306  }
307  /* XOR entire buf and send to out */
308  for (i = 0; i < 16; ++i) out[i] = in[i] ^ buf[i];
309  inlen -= 16;
310  out += 16;
311  in += 16;
312  }
313 }
