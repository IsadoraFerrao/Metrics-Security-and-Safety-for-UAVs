298  /**
299  Initialize the Blowfish block cipher
300  @param key The symmetric key you wish to pass
301  @param keylen The key length in bytes
302  @param num_rounds The number of rounds desired (0 for default)
303  @param skey The key in as scheduled by this function.
304  @return CRYPT_OK if successful
305  */
306 int blowfish_setup(const unsigned char *key, int keylen, int num_rounds,
307  symmetric_key *skey)
308 {
309  ulong32 x, y, z, A;
310  unsigned char B[8];
311 
312  LTC_ARGCHK(key != NULL);
313  LTC_ARGCHK(skey != NULL);
314 
315  /* check key length */
316  if (keylen < 8 || keylen > 56) {
317  return CRYPT_INVALID_KEYSIZE;
318  }
319 
320  /* check rounds */
321  if (num_rounds != 0 && num_rounds != 16) {
322  return CRYPT_INVALID_ROUNDS;
323  }
324 
325  /* load in key bytes (Supplied by David Hopwood) */
326  for (x = y = 0; x < 18; x++) {
327  A = 0;
328  for (z = 0; z < 4; z++) {
329  A = (A << 8) | ((ulong32)key[y++] & 255);
330  if (y == (ulong32)keylen) {
331  y = 0;
332  }
333  }
334  skey->blowfish.K[x] = ORIG_P[x] ^ A;
335  }
336 
337  /* copy sboxes */
338  for (x = 0; x < 4; x++) {
339  for (y = 0; y < 256; y++) {
340  skey->blowfish.S[x][y] = ORIG_S[x][y];
341  }
342  }
343 
344  /* encrypt K array */
345  for (x = 0; x < 8; x++) {
346  B[x] = 0;
347  }
348 
349  for (x = 0; x < 18; x += 2) {
350  /* encrypt it */
351  blowfish_ecb_encrypt(B, B, skey);
352  /* copy it */
353  LOAD32H(skey->blowfish.K[x], &B[0]);
354  LOAD32H(skey->blowfish.K[x+1], &B[4]);
355  }
356 
357  /* encrypt S array */
358  for (x = 0; x < 4; x++) {
359  for (y = 0; y < 256; y += 2) {
360  /* encrypt it */
361  blowfish_ecb_encrypt(B, B, skey);
362  /* copy it */
363  LOAD32H(skey->blowfish.S[x][y], &B[0]);
364  LOAD32H(skey->blowfish.S[x][y+1], &B[4]);
365  }
366  }
367 
368 #ifdef LTC_CLEAN_STACK
369  zeromem(B, sizeof(B));
370 #endif
371 
372  return CRYPT_OK;
373 }

382  Encrypts a block of text with Blowfish
383  @param pt The input plaintext (8 bytes)
384  @param ct The output ciphertext (8 bytes)
385  @param skey The key as scheduled
386  @return CRYPT_OK if successful

388 #ifdef LTC_CLEAN_STACK
389 static int _blowfish_ecb_encrypt(const unsigned char *pt, unsigned char *ct, const symmetric_key *skey)
390 #else
391 int blowfish_ecb_encrypt(const unsigned char *pt, unsigned char *ct, const symmetric_key *skey)
392 #endif
393 {
394  ulong32 L, R;
395  int r;
396 #ifndef __GNUC__
397  const ulong32 *S1, *S2, *S3, *S4;
398 #endif
399 
400  LTC_ARGCHK(pt != NULL);
401  LTC_ARGCHK(ct != NULL);
402  LTC_ARGCHK(skey != NULL);
403 
404 #ifndef __GNUC__
405  S1 = skey->blowfish.S[0];
406  S2 = skey->blowfish.S[1];
407  S3 = skey->blowfish.S[2];
408  S4 = skey->blowfish.S[3];
409 #endif
410 
411  /* load it */
412  LOAD32H(L, &pt[0]);
413  LOAD32H(R, &pt[4]);
414 
415  /* do 16 rounds */
416  for (r = 0; r < 16; ) {
417  L ^= skey->blowfish.K[r++]; R ^= F(L);
418  R ^= skey->blowfish.K[r++]; L ^= F(R);
419  L ^= skey->blowfish.K[r++]; R ^= F(L);
420  R ^= skey->blowfish.K[r++]; L ^= F(R);
421  }
422 
423  /* last keying */
424  R ^= skey->blowfish.K[17];
425  L ^= skey->blowfish.K[16];
426 
427  /* store */
428  STORE32H(R, &ct[0]);
429  STORE32H(L, &ct[4]);
430 
431  return CRYPT_OK;
