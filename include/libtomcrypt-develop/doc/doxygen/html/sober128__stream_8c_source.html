53 static void cycle(ulong32 *R)
54 {
55  ulong32 t;
56  int i;
57 
58  STEP(R,0);
59  t = R[0];
60  for (i = 1; i < N; ++i) {
61  R[i-1] = R[i];
62  }
63  R[N-1] = t;
64 }

159 int sober128_stream_setup(sober128_state *st, const unsigned char *key, unsigned long keylen)
160 {
161  ulong32 i, k;
162 
163  LTC_ARGCHK(st != NULL);
164  LTC_ARGCHK(key != NULL);
165  LTC_ARGCHK(keylen > 0);
166 
167  /* keylen must be multiple of 4 bytes */
168  if ((keylen & 3) != 0) {
169  return CRYPT_INVALID_KEYSIZE;
170  }
171 
172  /* Register initialised to Fibonacci numbers */
173  st->R[0] = 1;
174  st->R[1] = 1;
175  for (i = 2; i < N; ++i) {
176  st->R[i] = st->R[i-1] + st->R[i-2];
177  }
178  st->konst = INITKONST;
179 
180  for (i = 0; i < keylen; i += 4) {
181  k = BYTE2WORD((unsigned char *)&key[i]);
182  ADDKEY(k);
183  cycle(st->R);
184  XORNL(nltap(st));
185  }
186 
187  /* also fold in the length of the key */
188  ADDKEY(keylen);
189 
190  /* now diffuse */
191  s128_diffuse(st);
192  s128_genkonst(st);
193  s128_savestate(st);
194  st->nbuf = 0;
195 
196  return CRYPT_OK;
197 }

252 {

254

258

262  st->sbuf >>= 8;

264  --inlen;
