75 /* a simple macro for making hash "process" functions */
76 #define HASH_PROCESS(func_name, compress_name, state_var, block_size) \
77 int func_name (hash_state * md, const unsigned char *in, unsigned long inlen) \
78 { \
79  unsigned long n; \
80  int err; \
81  LTC_ARGCHK(md != NULL); \
82  LTC_ARGCHK(in != NULL); \
83  if (md-> state_var .curlen > sizeof(md-> state_var .buf)) { \
84  return CRYPT_INVALID_ARG; \
85  } \
86  if ((md-> state_var .length + inlen) < md-> state_var .length) { \
87  return CRYPT_HASH_OVERFLOW; \
88  } \
89  while (inlen > 0) { \
90  if (md-> state_var .curlen == 0 && inlen >= block_size) { \
91  if ((err = compress_name (md, in)) != CRYPT_OK) { \
92  return err; \
93  } \
94  md-> state_var .length += block_size * 8; \
95  in += block_size; \
96  inlen -= block_size; \
97  } else { \
98  n = MIN(inlen, (block_size - md-> state_var .curlen)); \
99  XMEMCPY(md-> state_var .buf + md-> state_var.curlen, in, (size_t)n); \
100  md-> state_var .curlen += n; \
101  in += n; \
102  inlen -= n; \
103  if (md-> state_var .curlen == block_size) { \
104  if ((err = compress_name (md, md-> state_var .buf)) != CRYPT_OK) { \
105  return err; \
106  } \
107  md-> state_var .length += 8*block_size; \
108  md-> state_var .curlen = 0; \
109  } \
110  } \
111  } \
112  return CRYPT_OK; \
113 }
